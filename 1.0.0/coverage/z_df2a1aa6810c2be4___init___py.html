182async def are_plugins_enabled_shared() -> bool: 

183 """Return the shared plugin-subsystem toggle, cached briefly to stay off the hot path.""" 

184 global _shared_enabled_cache 

185 async with _shared_enabled_cache_lock: 

186 cache = _shared_enabled_cache 

187 now = asyncio.get_event_loop().time() 

188 if cache is not None and (now - cache[1]) < _SHARED_ENABLED_CACHE_TTL: 

189 return cache[0] 

190 

191 value = await _read_shared_enabled() 

192 _shared_enabled_cache = (value, now) 

193 return value

212async def enable_plugins_shared(toggle: bool) -> bool: 

213 """Persist the global toggle to Redis (no TTL — operator kill-switch persists) and broadcast the change. 

214 

215 Returns True when Redis accepted the write. Falls back to the in-memory flag 

216 on Redis failure so the current worker still honors the operator intent. 

217 """ 

218 global _PLUGINS_ENABLED 

219 _PLUGINS_ENABLED = toggle 

220 _invalidate_shared_enabled_cache() 

221 

222 try: 

223 client = await _redis() 

224 except Exception as exc: 

225 _logger.warning("Failed to obtain Redis client for plugin toggle (%s) — change is local only", exc) 

226 return False 

227 if client is None: 

228 _logger.warning("Redis unavailable — plugin toggle change is local only") 

229 return False 

230 

231 try: 

232 await client.set(_REDIS_PLUGINS_ENABLED_KEY, "true" if toggle else "false") 

233 except Exception as exc: 

234 _logger.warning("Failed to write plugin toggle to Redis (%s) — change is local only", exc) 

235 return False 

236 

237 published = await _publish_invalidation({"type": "global_toggle", "enabled": toggle}) 

238 if not published: 

239 # Partial write: SET persisted, broadcast didn't. Peers will converge 

240 # on the next cache-TTL read (a few seconds) rather than instantly — 

241 # surface this at ERROR so it reaches on-call, not just DEBUG logs. 

242 _logger.error("Plugin global toggle persisted to Redis but broadcast failed — peer workers will lag until next cache refresh") 

243 return True

246async def _publish_invalidation(message: dict[str, Any]) -> bool: 

247 """Best-effort publish to the invalidation channel; logs but does not raise. 

248 

249 Listeners also evict on their own TTL, so a publish miss only widens the 

250 propagation window — it does not corrupt state. 

251 """ 

252 try: 

253 client = await _redis() 

254 except Exception as exc: 

255 _logger.warning("Plugin invalidation publish skipped — Redis client error (%s)", exc) 

256 return False 

257 if client is None: 

258 return False 

259 

260 try: 

261 await client.publish(_REDIS_INVALIDATION_CHANNEL, json.dumps(message)) 

262 return True 

263 except Exception as exc: 

264 _logger.warning("Plugin invalidation publish failed for %s (%s)", message.get("type"), exc) 

265 return False

280async def publish_plugin_mode_change(plugin_name: str, mode: str) -> bool: 

281 """Persist a per-plugin mode override locally and attempt to broadcast it. 

282 

283 Always updates the in-process override map so the change takes effect on 

284 this worker even when Redis is unavailable (single-node deployments). 

285 Returns True only when Redis accepted the write — the caller uses this 

286 signal to distinguish "cluster-wide" from "this worker only". 

287 

288 When the Redis SET succeeds, the local entry carries an expiry aligned 

289 with the Redis 24 h TTL so both copies clear together. When the Redis 

290 SET fails, the local entry is durable (no expiry) because the operator 

291 never got confirmation that a timer had started. 

292 """ 

293 try: 

294 client = await _redis() 

295 except Exception as exc: 

296 _logger.warning("Plugin mode Redis write skipped — client error (%s)", exc) 

297 _state.set_local_mode_override(plugin_name, mode, None) 

298 return False 

299 if client is None: 

300 _state.set_local_mode_override(plugin_name, mode, None) 

301 return False 

302 

303 try: 

304 await client.set(f"plugin:{plugin_name}:mode", mode, ex=_PLUGIN_MODE_TTL_SECONDS) 

305 except Exception as exc: 

306 _logger.warning("Plugin mode Redis SET failed for %s (%s)", plugin_name, exc) 

307 _state.set_local_mode_override(plugin_name, mode, None) 

308 return False 

309 

310 _state.set_local_mode_override(plugin_name, mode, time.monotonic() + _PLUGIN_MODE_TTL_SECONDS) 

311 # Peers use ttl_seconds to stamp their local copy with the same absolute 

312 # deadline — without it each worker would anchor expiry to its own 

313 # reception time and drift past the Redis key. 

314 published = await _publish_invalidation({"type": "mode_change", "plugin": plugin_name, "mode": mode, "ttl_seconds": _PLUGIN_MODE_TTL_SECONDS}) 

315 if not published: 

316 _logger.error( 

317 "Plugin mode override persisted to Redis for %s but broadcast failed — peer workers will lag until next cache refresh", 

318 plugin_name, 

319 ) 

320 return True

530async def _handle_invalidation_message(message: dict[str, Any]) -> None: 

531 """Dispatch one pub/sub frame to the matching invalidation handler. 

532 

533 Args: 

534 message: Raw redis-py pub/sub frame with ``type`` and ``data`` keys. 

535 """ 

536 if message.get("type") != "message": 

537 return # subscribe/unsubscribe notifications — not an invalidation event 

538 

539 try: 

540 data = json.loads(message["data"]) 

541 except (ValueError, KeyError, TypeError) as exc: 

542 _logger.warning("Ignoring malformed plugin invalidation message (%s)", exc) 

543 return 

544 

545 try: 

546 frame = _invalidation_adapter.validate_python(data) 

547 except _ValidationError as exc: 

548 _logger.warning("Ignoring unrecognised plugin invalidation frame %r (%s)", data, exc) 

549 return 

550 

551 match frame: 

552 case _GlobalToggleMsg(enabled=enabled): 

553 global _PLUGINS_ENABLED 

554 _PLUGINS_ENABLED = enabled 

555 _invalidate_shared_enabled_cache() 

556 _logger.debug("Pub/sub: global toggle set to %s", _PLUGINS_ENABLED) 

557 

558 case _ModeChangeMsg(plugin=plugin, mode=mode, ttl_seconds=ttl_seconds): 

559 # Mirror the broadcast into the local override map so this worker's 

560 # rebuild uses the new mode even if the Redis MGET on rebuild races 

561 # with the key's 24 h TTL. ``ttl_seconds`` arrives from the publisher 

562 # so every peer uses the same absolute deadline regardless of when 

563 # the broadcast lands locally. 

564 _state.set_local_mode_override(plugin, mode, time.monotonic() + ttl_seconds) 

565 await invalidate_all_plugin_managers() 

566 _logger.debug("Pub/sub: mode change for %s, all managers invalidated", plugin) 

567 

568 case _TeamBindingChangeMsg(team_id=team_id) if _plugin_manager_factory is not None: 

569 try: 

570 await _plugin_manager_factory.invalidate_team(team_id) 

571 _logger.debug("Pub/sub: team_binding_change, invalidated team %s", team_id) 

572 except Exception as exc: 

573 _logger.warning("Pub/sub team_binding_change failed for %s (%s)", team_id, exc) 

574 

575 case _BindingChangeMsg(context_id=context_id) if _plugin_manager_factory is not None: 

576 try: 

577 await _plugin_manager_factory.reload_tenant(context_id) 

578 _logger.debug("Pub/sub: binding change, reloaded context %s", context_id) 

579 except Exception as exc: 

580 _logger.warning("Pub/sub binding_change reload failed for %s (%s)", context_id, exc)

583async def _plugin_invalidation_listener() -> None: 

584 """Subscribe to the invalidation channel and dispatch messages until cancelled. 

585 

586 Reconnects on transport errors with exponential backoff plus jitter; the 

587 TTL-based cache refresh bounds staleness if this listener stays down. 

588 """ 

589 backoff = 1.0 

590 max_backoff = 30.0 

591 consecutive_failures = 0 

592 

593 while True: 

594 try: 

595 client = await _redis() 

596 if not client: 

597 _logger.debug("Plugin invalidation listener: Redis unavailable, retrying in 10s") 

598 await asyncio.sleep(10) 

599 continue 

600 

601 pubsub = client.pubsub() 

602 await pubsub.subscribe(_REDIS_INVALIDATION_CHANNEL) 

603 _logger.info("Plugin invalidation listener subscribed to %s", _REDIS_INVALIDATION_CHANNEL) 

604 backoff = 1.0 

605 consecutive_failures = 0 

606 

607 async for message in pubsub.listen(): 

608 await _handle_invalidation_message(message) 

609 

610 except asyncio.CancelledError: 

611 _logger.info("Plugin invalidation listener cancelled") 

612 break 

613 except Exception as exc: 

614 consecutive_failures += 1 

615 level = logging.ERROR if consecutive_failures >= 5 else logging.WARNING 

616 jitter = random.uniform(0, backoff / 2) # nosec B311 - jitter for backoff, not cryptographic 

617 delay = min(max_backoff, backoff + jitter) 

618 _logger.log( 

619 level, 

620 "Plugin invalidation listener error (%s, failure #%d), reconnecting in %.1fs", 

621 exc, 

622 consecutive_failures, 

623 delay, 

624 ) 

625 await asyncio.sleep(delay) 

626 backoff = min(max_backoff, backoff * 2)

629async def start_plugin_invalidation_listener() -> None: 

630 """Start the pub/sub listener if a Redis provider is wired and the task is not already running. 

631 

632 Single-node deployments without Redis would otherwise spin the listener's 

633 10-second retry loop indefinitely. Skip the task entirely in that case — 

634 operators can register a provider later and call this function again. 

635 Concurrent callers are serialized via ``_pubsub_start_lock`` so only one 

636 task is ever created. 

637 """ 

638 global _pubsub_task 

639 async with _pubsub_start_lock: 

640 if _pubsub_task is not None and not _pubsub_task.done(): 

641 return 

642 try: 

643 client = await _redis() 

644 except Exception as exc: # noqa: BLE001 — probe failure; log and bail 

645 _logger.warning("Plugin invalidation listener probe failed (%s); listener not started", exc) 

646 return 

647 if client is None: 

648 _logger.info("Plugin invalidation listener not started — no Redis provider registered") 

649 return 

650 _pubsub_task = asyncio.create_task(_plugin_invalidation_listener())

653async def stop_plugin_invalidation_listener() -> None: 

654 """Cancel the pub/sub listener and await its teardown.""" 

655 global _pubsub_task 

656 task = _pubsub_task 

657 _pubsub_task = None 

658 if task is None or task.done(): 

659 return 

660 task.cancel() 

661 try: 

662 await task 

663 except asyncio.CancelledError: 

664 pass
