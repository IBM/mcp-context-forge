1# -*- coding: utf-8 -*- 

2"""Location: ./mcpgateway/transports/mcp_ingress_mount.py 

3Copyright 2026 

4SPDX-License-Identifier: Apache-2.0 

5Authors: Mihai Criveti 

6 

7Swappable mount point for the public ``/mcp`` ingress. 

8 

9The mount itself owns no routing policy. It holds a registry of named 

10ingress ASGI apps and a callable that returns the name of whichever one 

11should serve the current request. Adding a new ingress (rust-public-proxy, 

12shadow-comparison, percentage-traffic-split, …) is a one-line 

13``register()`` call; selection policy is one function. 

14 

15This replaces the role of the prior ``MCPStreamableHTTPModeDispatcher`` 

16(which was hard-coded to a Python transport plus a single Rust transport, 

17with the per-request choice baked into its ``handle_streamable_http`` 

18method). See 

19``docs/docs/architecture/adr/051-swappable-mcp-ingress-mount.md`` for 

20the design decision and migration notes. 

21""" 

22 

23# Future 

24from __future__ import annotations 

25 

26# Standard 

27import logging 

28from typing import Awaitable, Callable, Dict, List, Optional, Protocol 

29 

30logger = logging.getLogger(__name__) 

31 

32 

33class ASGIApp(Protocol): 

34 """ASGI 3.0 callable. Any ingress registered with the mount must satisfy this.""" 

35 

36 async def __call__(self, scope: dict, receive: Callable[[], Awaitable[dict]], send: Callable[[dict], Awaitable[None]]) -> None: 

37 """ASGI 3.0 entry point. Implementations consume ``receive`` and emit via ``send``.""" 

38 

39 

40# An ``IngressSelector`` returns the registered ingress name to serve THIS 

41# request. It receives the ASGI scope so policies can inspect method, path, 

42# headers, query — useful for routing GET /mcp (SSE) differently from POST. 

43IngressSelector = Callable[[dict], str] 

44 

45 

46class MCPIngressMount: 

47 """Indirection at /mcp that delegates to a swappable inner ASGI app. 

48 

49 Lifecycle: 

50 1. Construct with a selector and (optionally) a fallback ingress. 

51 2. Register every ingress this build supports. 

52 3. Mount once at /mcp via ``app.mount("/mcp", mount.dispatch)``. 

53 4. Add/replace ingresses or swap the selector at runtime; the mount 

54 picks them up on the next request. 

55 

56 Concurrency: ingress lookup is a single dict read on the request hot 

57 path, no locks. Registration writes are not synchronized — the 

58 assumption is that ingresses are registered at startup and changed 

59 only via admin actions that already go through their own coordination 

60 (e.g. the runtime override coordinator). If you need atomic policy 

61 swaps mid-flight, swap the selector via :meth:`set_selector` rather 

62 than rewriting registrations. 

63 

64 Drain semantics: an ingress that owns long-lived connections (SSE, 

65 chunked POST) keeps serving them even after it is unregistered or 

66 selected away from. The next request just routes to the new ingress. 

67 This matches the prior dispatcher's "natural drain" property. 

68 """ 

69 

70 def __init__( 

71 self, 

72 *, 

73 selector: IngressSelector, 

74 fallback: Optional[ASGIApp] = None, 

75 ) -> None: 

76 """Initialize the mount with a selection policy and optional fallback. 

77 

78 Args: 

79 selector: Function that returns the registered ingress name to 

80 serve a given ASGI scope. 

81 fallback: ASGI app used when the selector returns a name that 

82 isn't registered. ``None`` means "send a 503 naming the 

83 missing ingress" — useful in dev to fail loudly; production 

84 deployments typically pass the Python transport here. 

85 """ 

86 self._ingresses: Dict[str, ASGIApp] = {} 

87 self._selector = selector 

88 self._fallback = fallback 

89 

90 # ── registry ----------------------------------------------------------- 

91 

92 def register(self, name: str, app: ASGIApp) -> None: 

93 """Register an ingress under ``name``. 

94 

95 Idempotent — calling ``register("x", app1)`` then 

96 ``register("x", app2)`` replaces ``app1``. 

97 

98 Args: 

99 name: Identifier the selector will return to pick this app. 

100 app: ASGI 3.0 callable. 

101 """ 

102 self._ingresses[name] = app 

103 

104 def unregister(self, name: str) -> Optional[ASGIApp]: 

105 """Remove an ingress; returns the prior app for graceful shutdown. 

106 

107 Args: 

108 name: Ingress identifier. 

109 

110 Returns: 

111 The previously-registered ASGI app, or ``None`` if the name 

112 wasn't registered. Caller is responsible for any drain or 

113 cleanup on the returned app. 

114 """ 

115 return self._ingresses.pop(name, None) 

116 

117 def names(self) -> List[str]: 

118 """Return the sorted list of currently-registered ingress names. 

119 

120 Returns: 

121 Sorted list of ingress names. Useful for diagnostics, /health, 

122 and the 503 body when a selector returns an unknown name. 

123 """ 

124 return sorted(self._ingresses) 

125 

126 # ── policy ------------------------------------------------------------ 

127 

128 def set_selector(self, selector: IngressSelector) -> None: 

129 """Atomically swap the selection policy. 

130 

131 Args: 

132 selector: New selector function. Takes effect on the next 

133 request — in-flight requests already inside an ingress 

134 continue under the old selection. 

135 """ 

136 self._selector = selector 

137 

138 def set_fallback(self, app: Optional[ASGIApp]) -> None: 

139 """Update the fallback app used when the selector misses. 

140 

141 Args: 

142 app: New fallback ASGI app, or ``None`` to restore the 

143 503-on-miss behavior. 

144 """ 

145 self._fallback = app 

146 

147 # ── ASGI app --------------------------------------------------------- 

148 

149 async def dispatch(self, scope: dict, receive: Callable, send: Callable) -> None: 

150 """ASGI entry point. Picks an ingress per request and delegates. 

151 

152 Args: 

153 scope: ASGI scope. 

154 receive: ASGI receive callable. 

155 send: ASGI send callable. 

156 """ 

157 name = self._selector(scope) 

158 app = self._ingresses.get(name) 

159 if app is None: 

160 # Selector miss is an invariant violation — the registered set 

161 # and the policy are out of sync. Either path (fallback or 

162 # 503) deserves a warning, not debug, so a misconfigured 

163 # selector doesn't silently route auth-sensitive traffic to 

164 # the wrong handler. 

165 if self._fallback is None: 

166 logger.warning( 

167 "MCPIngressMount: selector returned %r but no ingress is registered under that name; available=%s. Returning 503.", 

168 name, 

169 self.names(), 

170 ) 

171 await self._send_503(send, name) 

172 return 

173 logger.warning( 

174 "MCPIngressMount: selector returned %r which is not registered; available=%s. Falling back to the configured fallback app.", 

175 name, 

176 self.names(), 

177 ) 

178 app = self._fallback 

179 await app(scope, receive, send) 

180 

181 async def _send_503(self, send: Callable, missing_name: str) -> None: 

182 """Emit a 503 response naming the missing ingress. 

183 

184 Args: 

185 send: ASGI send callable. 

186 missing_name: The ingress name the selector returned. 

187 """ 

188 body = (f"MCP ingress {missing_name!r} is not registered for this build; " f"available: {self.names() or '[]'}").encode("utf-8") 

189 await send( 

190 { 

191 "type": "http.response.start", 

192 "status": 503, 

193 "headers": [ 

194 (b"content-type", b"text/plain; charset=utf-8"), 

195 (b"content-length", str(len(body)).encode("ascii")), 

196 ], 

197 } 

198 ) 

199 await send({"type": "http.response.body", "body": body, "more_body": False})
