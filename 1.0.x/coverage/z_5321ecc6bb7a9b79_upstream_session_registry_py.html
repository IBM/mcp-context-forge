7Upstream MCP session registry (1:1 per downstream session). 

8Replaces the previous ``SessionAffinity`` which keyed upstream sessions by 

9``(user_identity, url, identity_hash, transport_type, gateway_id)``. That 

10sharing leaked state between downstream MCP sessions whose callers happened 

11to share the same identity (issue #4205): two chat tabs opened by the same 

12user would receive the same counter state, because both were wired to the 

13same pooled upstream session. 

14This registry enforces 1:1 binding between a downstream MCP session (as 

15identified by its ``Mcp-Session-Id``) and each upstream gateway it talks to. 

16Within one downstream session, tool calls still reuse a single upstream 

17session per gateway — so the per-call latency win of pooling survives — but 

18nothing is ever shared across downstream sessions. 

19"""

188# # 

189# Detecting "transport is broken but nobody told us" requires reaching into # 

190# ClientSession's private anyio streams. That coupling is fragile across MCP # 

191# SDK versions, so this is the ONE place allowed to touch those internals. #

198_MCP_SDK_TRANSPORT_PROBE_COMPATIBLE_VERSIONS = ">=1.27.0,<2.0.0"

200# One-shot guard for the SDK-drift log: WARNING on first occurrence per process 

201# (so operators can't miss "the SDK shape just changed under us") then DEBUG on 

202# every subsequent call (so a sustained mismatch doesn't flood logs — this probe 

203# runs on every acquire()). Mutable module state, not a constant — pylint's 

204# all-caps convention doesn't fit. 

205_sdk_drift_warning_emitted = False # pylint: disable=invalid-name

208def _mcp_transport_is_broken(session: ClientSession) -> bool: 

209 """Peek at a ``ClientSession``'s internal anyio streams to detect a dead transport. 

210 

211 Returns True only when we can positively confirm the transport is gone 

212 (closed write stream, or receive channels fully drained). Returns False on 

213 any ambiguity — including when SDK internals have shifted shape — so that 

214 callers degrade to owner-task liveness rather than evicting a session 

215 that might still be usable. 

216 

217 Validated MCP SDK range lives in 

218 ``_MCP_SDK_TRANSPORT_PROBE_COMPATIBLE_VERSIONS``; bump that marker after 

219 revalidating when the SDK changes. 

220 """ 

221 global _sdk_drift_warning_emitted # pylint: disable=global-statement 

222 try: 

223 write_stream = getattr(session, "_write_stream", None) 

224 if write_stream is None: 

225 return False 

226 if getattr(write_stream, "_closed", False) is True: 

227 return True 

228 state = getattr(write_stream, "_state", None) 

229 if state is None: 

230 return False 

231 open_rx = getattr(state, "open_receive_channels", 1) 

232 if isinstance(open_rx, int) and open_rx == 0: 

233 return True 

234 except Exception as exc: # noqa: BLE001 — degrade gracefully if MCP internals shift 

235 if not _sdk_drift_warning_emitted: 

236 _sdk_drift_warning_emitted = True 

237 logger.warning( 

238 "MCP transport-broken probe raised %s: %s; validated against SDK %s. Next acquires will fall back to owner-task liveness only; subsequent probe failures logged at DEBUG.", 

239 type(exc).__name__, 

240 exc, 

241 _MCP_SDK_TRANSPORT_PROBE_COMPATIBLE_VERSIONS, 

242 ) 

243 else: 

244 logger.debug( 

245 "MCP transport-broken probe raised %s: %s; next acquire will fall back to owner-task liveness only", 

246 type(exc).__name__, 

247 exc, 

248 ) 

249 return False

252@dataclass 

253class UpstreamSession: 

254 """A single upstream MCP session bound to one downstream session. 

255 

256 The four identity fields (downstream_session_id, gateway_id, url, 

257 transport_type) are logically immutable after construction: the registry 

258 keys its session map on (downstream_session_id, gateway_id), and routing 

259 decisions depend on url + transport_type. Re-assigning any of them would 

260 leave the registry indexing a session under one key while the session 

261 itself thought it belonged to another — a split-brain that violates the 

262 1:1 invariant this class exists to enforce. ``__setattr__`` refuses such 

263 assignments at runtime so the invariant can't drift silently. The 

264 remaining fields (last_used, use_count, _closed, …) are mutable bookkeeping. 

265 """ 

266 

267 downstream_session_id: str 

268 gateway_id: str 

269 url: str 

270 transport_type: TransportType 

271 session: ClientSession 

272 created_at: float = field(default_factory=time.time) 

273 last_used: float = field(default_factory=time.time) 

274 use_count: int = 0 

275 _closed: bool = field(default=False, repr=False) 

276 _owner_task: Optional[asyncio.Task] = field(default=None, repr=False) 

277 _shutdown_event: Optional[asyncio.Event] = field(default=None, repr=False) 

278 

279 def __setattr__(self, name: str, value: Any) -> None: 

280 """Reject post-construction reassignment of identity fields.""" 

281 if name in _IDENTITY_FIELDS and name in self.__dict__: 

282 raise AttributeError(f"{name!r} is immutable on UpstreamSession after construction") 

283 super().__setattr__(name, value) 

284 

285 @property 

286 def idle_seconds(self) -> float: 

287 """Seconds since this session was last used.""" 

288 return time.time() - self.last_used 

289 

290 @property 

291 def age_seconds(self) -> float: 

292 """Seconds since this session was created.""" 

293 return time.time() - self.created_at 

294 

295 @property 

296 def is_closed(self) -> bool: 

297 """Whether the session has been marked closed or its transport has broken.""" 

298 if self._closed: 

299 return True 

300 if self._owner_task is not None and self._owner_task.done(): 

301 return True 

302 return _mcp_transport_is_broken(self.session)

305async def _default_session_factory(req: SessionCreateRequest) -> tuple[ClientSession, Any]: 

306 """Owner-task wrapper that builds the real transport + ClientSession. 

307 

308 Runs inside a dedicated asyncio.Task so the transport's anyio cancel scope 

309 is bound to that task, not to whichever request handler happens to be 

310 making the acquire() call. If the request task is cancelled (client 

311 disconnect, timeout), the upstream transport is NOT torn down with it. 

312 """ 

313 if req.transport_type is TransportType.SSE: 

314 if req.httpx_client_factory is not None: 

315 transport_ctx = sse_client( 

316 url=req.url, 

317 headers=req.headers, 

318 httpx_client_factory=req.httpx_client_factory, 

319 timeout=req.timeout_seconds, 

320 ) 

321 else: 

322 transport_ctx = sse_client( 

323 url=req.url, 

324 headers=req.headers, 

325 timeout=req.timeout_seconds, 

326 ) 

327 else: 

328 if req.httpx_client_factory is not None: 

329 transport_ctx = streamablehttp_client( 

330 url=req.url, 

331 headers=req.headers, 

332 httpx_client_factory=req.httpx_client_factory, 

333 timeout=req.timeout_seconds, 

334 ) 

335 else: 

336 transport_ctx = streamablehttp_client( 

337 url=req.url, 

338 headers=req.headers, 

339 timeout=req.timeout_seconds, 

340 ) 

341 

342 shutdown_event = asyncio.Event() 

343 loop = asyncio.get_running_loop() 

344 ready: asyncio.Future[tuple[ClientSession, Any]] = loop.create_future() 

345 

346 async def owner() -> None: 

347 """Own the transport + ClientSession lifecycle; unblock on shutdown_event.""" 

348 try: 

349 async with transport_ctx as streams: 

350 read_stream, write_stream = streams[0], streams[1] 

351 message_handler = None 

352 if req.message_handler_factory is not None: 

353 try: 

354 message_handler = req.message_handler_factory( 

355 req.url, 

356 req.gateway_id, 

357 downstream_session_id=req.downstream_session_id, 

358 ) 

359 except Exception as exc: # noqa: BLE001 — handler failure is not fatal 

360 logger.warning( 

361 "Failed to build message handler for %s: %s", 

362 sanitize_url_for_logging(req.url), 

363 exc, 

364 ) 

365 async with ClientSession(read_stream, write_stream, message_handler=message_handler) as session: 

366 await session.initialize() 

367 if not ready.done(): 

368 ready.set_result((session, transport_ctx)) 

369 # Block until the registry signals shutdown; do NOT rely on 

370 # task cancellation from a request handler (see class docs). 

371 await shutdown_event.wait() 

372 except Exception as exc: # noqa: BLE001 — see below 

373 # Broad catch on purpose: the upstream-setup path runs many 

374 # third-party coroutines (httpx, anyio, MCP SDK) whose exception 

375 # classes we cannot enumerate. BaseException is deliberately NOT 

376 # caught — SystemExit / KeyboardInterrupt / CancelledError must 

377 # propagate so the task exits promptly during shutdown. 

378 if not ready.done(): 

379 ready.set_exception(RuntimeError(f"Failed to create upstream MCP session for {req.url}: {exc}")) 

380 

381 task = asyncio.create_task(owner(), name=f"upstream-session-{sanitize_url_for_logging(req.url)}") 

382 

383 def _log_owner_exit(done_task: asyncio.Task) -> None: 

384 """Surface unexpected owner-task deaths so an orphaned upstream session is visible to ops.""" 

385 if done_task.cancelled(): 

386 return 

387 exc = done_task.exception() 

388 if exc is not None: 

389 logger.warning( 

390 "Upstream MCP owner task for %s exited with %s: %s — upstream session may be orphaned", 

391 sanitize_url_for_logging(req.url), 

392 type(exc).__name__, 

393 exc, 

394 ) 

395 

396 task.add_done_callback(_log_owner_exit) 

397 

398 success = False 

399 try: 

400 session, transport_ctx_ref = await asyncio.wait_for(ready, timeout=req.timeout_seconds) 

401 success = True 

402 finally: 

403 if not success: 

404 shutdown_event.set() 

405 task.cancel() 

406 with anyio.move_on_after(_DEFAULT_SHUTDOWN_TIMEOUT_SECONDS): 

407 try: 

408 await task 

409 except asyncio.CancelledError: 

410 # The owner task itself got cancelled during cleanup — expected after task.cancel(). 

411 pass 

412 except Exception as exc: # noqa: BLE001 — cleanup unwind after failed ready # pragma: no cover 

413 # Defensive: the owner's own `except Exception` swallows all Exception 

414 # subclasses before they can escape the task, so reaching this branch 

415 # would require a BaseException that slipped through — unreachable 

416 # in practice but kept to narrow against future refactors. 

417 logger.debug( 

418 "Owner-task cleanup after failed session create raised %s: %s", 

419 type(exc).__name__, 

420 exc, 

421 ) 

422 

423 # Smuggle the owner task + shutdown event onto the ClientSession object so 

424 # _create_session() (which only gets back `(session, transport_ctx)` from 

425 # the factory) can recover them without a wider factory return contract. 

426 # Tests that replace the factory must mirror this convention. 

427 setattr(session, "_cf_owner_task", task) # type: ignore[attr-defined] 

428 setattr(session, "_cf_shutdown_event", shutdown_event) # type: ignore[attr-defined] 

429 return session, transport_ctx_ref

432class UpstreamSessionRegistry: 

433 """Maps ``(downstream_session_id, gateway_id)`` to a single upstream session. 

434 

435 Isolation is structural: two downstream sessions cannot share an upstream 

436 session, even when they carry the same user identity. Within one downstream 

437 session, concurrent calls to acquire() for the same gateway return the same 

438 underlying MCP ClientSession — connection reuse stays intact. 

439 

440 Lifetime of an upstream session follows the downstream session: callers 

441 evict via evict_session() on DELETE /mcp (or session expiry). There is no 

442 wall-clock TTL; staleness is handled by a health probe on reuse after the 

443 idle validation window. 

444 

445 The registry is purely in-process. Multi-worker stickiness for a given 

446 downstream session is provided by the session-affinity layer in 

447 streamablehttp_transport; the registry on each worker only sees requests 

448 that have already been routed to the owning worker. 

449 """ 

450 

451 def __init__( 

452 self, 

453 *, 

454 session_factory: Optional[SessionFactory] = None, 

455 message_handler_factory: Optional[MessageHandlerFactory] = None, 

456 idle_validation_seconds: float = _DEFAULT_IDLE_VALIDATION_SECONDS, 

457 health_check_timeout_seconds: float = _DEFAULT_HEALTH_CHECK_TIMEOUT_SECONDS, 

458 session_create_timeout_seconds: float = _DEFAULT_SESSION_CREATE_TIMEOUT_SECONDS, 

459 shutdown_timeout_seconds: float = _DEFAULT_SHUTDOWN_TIMEOUT_SECONDS, 

460 ) -> None: 

461 """Build a registry. ``session_factory`` is injectable for tests.""" 

462 self._session_factory: SessionFactory = session_factory or _default_session_factory 

463 self._message_handler_factory = message_handler_factory 

464 self._idle_validation_seconds = idle_validation_seconds 

465 self._health_check_timeout_seconds = health_check_timeout_seconds 

466 self._session_create_timeout_seconds = session_create_timeout_seconds 

467 self._shutdown_timeout_seconds = shutdown_timeout_seconds 

468 

469 self._sessions: dict[tuple[str, str], UpstreamSession] = {} 

470 self._key_locks: dict[tuple[str, str], asyncio.Lock] = {} 

471 self._global_lock = asyncio.Lock() 

472 self._metrics = _RegistryMetrics()

478 @asynccontextmanager 

479 async def acquire( 

480 self, 

481 *, 

482 downstream_session_id: str, 

483 gateway_id: str, 

484 url: str, 

485 headers: Optional[dict[str, str]], 

486 transport_type: TransportType, 

487 httpx_client_factory: Optional[HttpxClientFactory] = None, 

488 ) -> AsyncIterator[UpstreamSession]: 

489 """Get or create the upstream session for ``(downstream_session_id, gateway_id)``. 

490 

491 Concurrent acquires for the same key serialize on the per-key lock only 

492 across the check-or-create phase. Once a session is chosen, the lock is 

493 released and the yielded ClientSession multiplexes concurrent requests 

494 over its transport via MCP request ids. 

495 """ 

496 if not downstream_session_id: 

497 raise ValueError("downstream_session_id is required; caller must pin an upstream session to a downstream one") 

498 if not gateway_id: 

499 raise ValueError("gateway_id is required") 

500 

501 key = (downstream_session_id, gateway_id) 

502 key_lock = await self._get_key_lock(key) 

503 

504 async with key_lock: 

505 session = self._sessions.get(key) 

506 reason = _AcquireDecision.REUSE 

507 

508 if session is None or session.is_closed: 

509 reason = _AcquireDecision.CREATE 

510 elif session.idle_seconds > self._idle_validation_seconds: 

511 healthy = await self._probe_health(session) 

512 if not healthy: 

513 logger.info( 

514 "Upstream session health probe failed, recreating (gateway=%s)", 

515 gateway_id, 

516 ) 

517 await self._close_session(session) 

518 self._sessions.pop(key, None) 

519 self._metrics.health_check_recreates += 1 

520 reason = _AcquireDecision.CREATE 

521 

522 if reason is _AcquireDecision.CREATE: 

523 session = await self._create_session( 

524 downstream_session_id=downstream_session_id, 

525 gateway_id=gateway_id, 

526 url=url, 

527 headers=headers, 

528 transport_type=transport_type, 

529 httpx_client_factory=httpx_client_factory, 

530 ) 

531 self._sessions[key] = session 

532 self._metrics.creates += 1 

533 else: 

534 self._metrics.reuses += 1 

535 

536 if session is None: # nosec B101 - runtime check instead of assert 

537 raise RuntimeError("Session unexpectedly None after acquire") 

538 session.last_used = time.time() 

539 session.use_count += 1 

540 

541 # Hand out the session with no lock held: MCP ClientSession multiplexes 

542 # concurrent requests over its transport via JSON-RPC ids, so there's no 

543 # reason to serialize callers. If the caller's body raises a transport- 

544 # level error (server closed the stream, socket broke), evict so the 

545 # next acquire rebuilds instead of handing out a dead session. 

546 try: 

547 yield session 

548 except (OSError, anyio.ClosedResourceError, anyio.BrokenResourceError) as exc: 

549 logger.info( 

550 "acquire() caller raised %s for gateway=%s; evicting upstream so next acquire rebuilds", 

551 type(exc).__name__, 

552 gateway_id, 

553 ) 

554 await self._evict_key(key) 

555 raise 

556 # All other exceptions (tool-level errors from the upstream, caller 

557 # application errors) intentionally leave the session in place — the 

558 # transport is fine, the caller just didn't like the result. 

559 

560 async def evict_session(self, downstream_session_id: str) -> int: 

561 """Close and remove every upstream session owned by this downstream session id. 

562 

563 Returns the number of sessions evicted. Safe to call when no sessions 

564 exist. Evictions fire concurrently so a downstream session with many 

565 gateways doesn't block on the slowest-draining upstream. 

566 """ 

567 async with self._global_lock: 

568 keys = [k for k in self._sessions if k[0] == downstream_session_id] 

569 return await self._evict_keys_in_parallel(keys) 

570 

571 async def evict_gateway(self, gateway_id: str) -> int: 

572 """Close and remove every upstream session pointing at a given gateway. 

573 

574 Intended for gateway-removal/rotation; downstream sessions survive but 

575 will rebuild upstream state on the next acquire() call. Evictions run 

576 concurrently — fires post-commit on admin update/delete, so a slow 

577 drain must not stall the response. 

578 """ 

579 async with self._global_lock: 

580 keys = [k for k in self._sessions if k[1] == gateway_id] 

581 return await self._evict_keys_in_parallel(keys) 

582 

583 async def _evict_keys_in_parallel(self, keys: list[tuple[str, str]]) -> int: 

584 """Run _evict_key concurrently for a set of keys; returns the count that succeeded.""" 

585 if not keys: 

586 return 0 

587 results = await asyncio.gather(*[self._evict_key(k) for k in keys], return_exceptions=True) 

588 return sum(1 for r in results if r is True)

590 async def close_all(self) -> None: 

591 """Drain every upstream session concurrently. Intended for app shutdown. 

592 

593 Each ``_evict_key`` can take up to ``shutdown_timeout_seconds`` waiting 

594 for the owner task to exit; running them in series on a worker with 

595 dozens of downstream sessions would turn shutdown into a multi-minute 

596 stall. ``asyncio.gather`` caps the total drain at roughly 

597 ``shutdown_timeout_seconds`` plus a small constant. 

598 """ 

599 async with self._global_lock: 

600 keys = list(self._sessions.keys()) 

601 if not keys: 

602 return 

603 await asyncio.gather(*[self._evict_key(k) for k in keys], return_exceptions=True)

620 async def _get_key_lock(self, key: tuple[str, str]) -> asyncio.Lock: 

621 """Return (creating if needed) the per-key lock used to serialize check-or-create.""" 

622 async with self._global_lock: 

623 lock = self._key_locks.get(key) 

624 if lock is None: 

625 lock = asyncio.Lock() 

626 self._key_locks[key] = lock 

627 return lock 

628 

629 async def _evict_key(self, key: tuple[str, str]) -> bool: 

630 """Close the session at ``key`` (if any) and drop its entry. Returns True on eviction.""" 

631 lock = await self._get_key_lock(key) 

632 async with lock: 

633 session = self._sessions.pop(key, None) 

634 if session is None: 

635 self._key_locks.pop(key, None) 

636 return False 

637 await self._close_session(session) 

638 self._metrics.evictions += 1 

639 # Drop the key lock after release; safe because we hold _global_lock 

640 # only briefly and any new acquire() will recreate the lock lazily. 

641 async with self._global_lock: 

642 self._key_locks.pop(key, None) 

643 return True

645 async def _create_session( 

646 self, 

647 *, 

648 downstream_session_id: str, 

649 gateway_id: str, 

650 url: str, 

651 headers: Optional[dict[str, str]], 

652 transport_type: TransportType, 

653 httpx_client_factory: Optional[HttpxClientFactory], 

654 ) -> UpstreamSession: 

655 """Build a fresh upstream session via the configured SessionFactory.""" 

656 merged_headers = {"Accept": "application/json, text/event-stream"}

660 # forwarded upstream. Upstream sees its own session id, which it 

661 # assigns via its initialize response. 

662 for hdr in list(merged_headers):

664 del merged_headers[hdr] 

665 

666 req = SessionCreateRequest( 

667 url=url, 

668 transport_type=transport_type, 

669 headers=merged_headers, 

670 gateway_id=gateway_id, 

671 downstream_session_id=downstream_session_id, 

672 httpx_client_factory=httpx_client_factory, 

673 message_handler_factory=self._message_handler_factory, 

674 timeout_seconds=self._session_create_timeout_seconds, 

675 ) 

676 session, _transport_ctx = await self._session_factory(req) 

677 owner_task = getattr(session, "_cf_owner_task", None) 

678 shutdown_event = getattr(session, "_cf_shutdown_event", None) 

679 return UpstreamSession( 

680 downstream_session_id=downstream_session_id, 

681 gateway_id=gateway_id,

683 transport_type=transport_type, 

684 session=session, 

685 _owner_task=owner_task, 

686 _shutdown_event=shutdown_event, 

687 )

689 async def _probe_health(self, upstream: UpstreamSession) -> bool: 

690 """Run the health check chain against an idle session. Returns False if all probes fail. 

691 

692 Exception policy: we ADVANCE on ``TimeoutError`` and on 

693 ``McpError(METHOD_NOT_FOUND)`` (the server chose not to implement 

694 this probe), and we FAIL FAST on everything else transport- or 

695 protocol-level (``OSError`` / anyio stream errors / other ``McpError``s) 

696 — recreating a session on "permission denied" or "request too large" 

697 would loop against the same failure. Genuinely unexpected exceptions 

698 (``AttributeError`` from SDK drift, etc.) propagate so they surface in 

699 telemetry instead of silently triggering a reconnect loop. 

700 """ 

701 for method in _HEALTH_CHECK_CHAIN: 

702 try: 

703 if method == "skip": 

704 return True 

705 with anyio.fail_after(self._health_check_timeout_seconds): 

706 if method == "ping": 

707 await upstream.session.send_ping() 

708 elif method == "list_tools": 

709 await upstream.session.list_tools() 

710 elif method == "list_prompts": 

711 await upstream.session.list_prompts() 

712 elif method == "list_resources": 

713 await upstream.session.list_resources() 

714 return True 

715 except McpError as exc: 

716 if exc.error.code == _METHOD_NOT_FOUND: 

717 continue # Server doesn't support this probe; try the next one. 

718 self._metrics.health_check_failures += 1 

719 return False 

720 except TimeoutError: 

721 continue 

722 except OSError: 

723 # Socket / stream error — upstream is dead. 

724 self._metrics.health_check_failures += 1 

725 return False 

726 self._metrics.health_check_failures += 1 

727 return False

729 async def _close_session(self, upstream: UpstreamSession) -> None: 

730 """Signal the owner task to shut down and wait, with a timeout fallback. 

731 

732 Uses ``asyncio.wait`` rather than ``await task`` so a rogue owner 

733 that catches CancelledError without re-raising cannot hang shutdown. 

734 The caller's cancellation cannot be used to interrupt ``await task`` 

735 in that case: asyncio chains the cancel into the awaited task, but 

736 if the awaited task refuses to die, the ``await`` is stuck waiting 

737 for it to complete. ``asyncio.wait`` returns once its own timer 

738 fires, regardless of the awaited task's state. 

739 

740 The leading-underscore fields on ``UpstreamSession`` are private to 

741 this module — the registry owns the session lifecycle and is the 

742 only legitimate mutator of ``_closed`` / ``_shutdown_event`` / 

743 ``_owner_task``. Pylint's ``protected-access`` rule is disabled 

744 inline for each access because the alternative (a public setter 

745 per field) would leak lifecycle mechanics to any caller that 

746 happened to import ``UpstreamSession``. 

747 """ 

748 # pylint: disable=protected-access 

749 if upstream._closed: 

750 return 

751 upstream._closed = True 

752 if upstream._shutdown_event is not None: 

753 upstream._shutdown_event.set() 

754 if upstream._owner_task is None or upstream._owner_task.done(): 

755 return 

756 

757 owner_task = upstream._owner_task 

758 # Give the task its graceful window — it should notice shutdown_event 

759 # and exit cleanly. 

760 done, _pending = await asyncio.wait({owner_task}, timeout=self._shutdown_timeout_seconds) 

761 if owner_task in done: 

762 # Task finished cleanly; consume any exception so asyncio doesn't 

763 # warn "Task exception was never retrieved". 

764 if not owner_task.cancelled(): 

765 exc = owner_task.exception() 

766 if exc is not None: 

767 logger.debug( 

768 "Owner task for %s (gateway=%s) exited during _close_session with %s: %s", 

769 sanitize_url_for_logging(upstream.url), 

770 upstream.gateway_id, 

771 type(exc).__name__, 

772 exc, 

773 ) 

774 return 

775 

776 # Grace period elapsed — force-cancel and give one more bounded wait. 

777 logger.warning( 

778 "Upstream session owner cleanup timed out for session=%s gateway=%s url=%s; force-cancelling", 

779 upstream.downstream_session_id, 

780 upstream.gateway_id, 

781 sanitize_url_for_logging(upstream.url), 

782 ) 

783 owner_task.cancel() 

784 done, _pending = await asyncio.wait({owner_task}, timeout=self._shutdown_timeout_seconds) 

785 if owner_task not in done: 

786 logger.warning( 

787 "Force-cancel of owner task did not complete within %.1fs for session=%s gateway=%s — task is orphaned but shutdown is proceeding", 

788 self._shutdown_timeout_seconds, 

789 upstream.downstream_session_id, 

790 upstream.gateway_id, 

791 ) 

792 return 

793 # Consume any final exception. 

794 if not owner_task.cancelled(): 

795 with contextlib.suppress(Exception): # noqa: BLE001 — final retrieval 

796 owner_task.result()
