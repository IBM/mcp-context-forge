117 def validate_user_password( 

118 self, 

119 password: str, 

120 email: Optional[str] = None, 

121 is_privileged: bool = False, 

122 ) -> bool: 

123 """Validate password for user accounts against comprehensive policy. 

124 

125 Args: 

126 password: Password to validate 

127 email: User's email address (for username-based validation) 

128 is_privileged: Whether this is a privileged account (requires longer password) 

129 

130 Returns: 

131 bool: True if password meets all requirements 

132 

133 Raises: 

134 PasswordPolicyError: If password doesn't meet requirements 

135 

136 Examples: 

137 >>> from mcpgateway.db import SessionLocal 

138 >>> with SessionLocal() as db: 

139 ... service = PasswordPolicyService(db) 

140 ... service.validate_user_password("SecureP@ssw0rd!", "alice@example.com") 

141 True 

142 """ 

143 if not password: 

144 raise PasswordPolicyError("Password is required") 

145 

146 # Minimum length: 12 characters (OWASP recommendation) 

147 # Privileged accounts: 22 characters (12 + 10 as recommended) 

148 min_length = getattr(settings, "password_min_length_user", 12) 

149 if is_privileged: 

150 min_length = getattr(settings, "password_min_length_privileged", 22) 

151 

152 if len(password) < min_length: 

153 raise PasswordPolicyError(f"Password must be at least {min_length} characters long " f"({'privileged account' if is_privileged else 'user account'})") 

154 

155 # Check complexity requirements (must have 3 of 4 character types) 

156 complexity_count = 0 

157 has_lower = bool(re.search(r"[a-z]", password)) 

158 has_upper = bool(re.search(r"[A-Z]", password)) 

159 has_digit = bool(re.search(r"[0-9]", password)) 

160 has_special = bool(re.search(r'[!@#$%^&*(),.?":{}|<>\-_=+\[\]\\;\'`~]', password)) 

161 

162 if has_lower: 

163 complexity_count += 1 

164 if has_upper: 

165 complexity_count += 1 

166 if has_digit: 

167 complexity_count += 1 

168 if has_special: 

169 complexity_count += 1 

170 

171 if complexity_count < 3: 

172 raise PasswordPolicyError("Password must contain at least 3 of the following: lowercase letters, uppercase letters, numbers, special characters") 

173 

174 # Check against common passwords 

175 is_common = password.lower() in COMMON_PASSWORDS 

176 if is_common: 

177 raise PasswordPolicyError("Password is too common or easily guessable") 

178 

179 # Check if password contains username 

180 if email: 

181 username = email.split("@")[0].lower() 

182 contains_username = len(username) >= 3 and username in password.lower() 

183 if contains_username: 

184 raise PasswordPolicyError("Password must not be based on your username") 

185 

186 # Check for sequential characters (e.g., "123", "abc") 

187 has_sequential = self._has_sequential_chars(password) 

188 if has_sequential: 

189 raise PasswordPolicyError("Password contains too many sequential characters") 

190 

191 return True

195 

196 Service accounts should use randomly generated passwords with minimum 

197 20 characters (approximately 130 bits of entropy with full keyboard charset). 

198 

199 Args: 

200 password: Password to validate

202 Returns: 

203 bool: True if password meets service account requirements 

204 

205 Raises: 

206 PasswordPolicyError: If password doesn't meet requirements 

207 

208 Examples: 

209 >>> from mcpgateway.db import SessionLocal 

210 >>> with SessionLocal() as db: 

211 ... service = PasswordPolicyService(db) 

212 ... # 20+ character random password 

213 ... service.validate_service_account_password("aB3$xY9#mN2@pQ7!wR5%") 

214 True 

215 """ 

216 if not password: 

217 raise PasswordPolicyError("Password is required") 

218 

219 # Service accounts: minimum 20 characters (recommended by pentesting report) 

220 min_length = getattr(settings, "password_min_length_service", 20) 

221 

222 if len(password) < min_length: 

223 raise PasswordPolicyError(f"Service account password must be at least {min_length} characters long " f"(recommended: use randomly generated passwords with 64-128 bits entropy)") 

224 

225 # Service accounts should have high entropy - check for randomness 

226 if not self._has_sufficient_entropy(password): 

227 raise PasswordPolicyError("Service account password appears to have low entropy. Use cryptographically secure random generation (e.g., UUID or secrets module)") 

228

231 async def check_password_history( 

232 self, 

233 email: str, 

234 new_password: str, 

235 history_count: int = 5, 

236 current_password_hash: Optional[str] = None, 

237 ) -> bool: 

238 """Check if password was used in recent history or matches current password. 

239 

240 Args: 

241 email: User's email address 

242 new_password: New password to check 

243 history_count: Number of previous passwords to check (default: 5) 

244 current_password_hash: Current password hash to check against (optional) 

245 

246 Returns: 

247 bool: True if password is not in history 

248 

249 Raises: 

250 PasswordPolicyError: If password was recently used or matches current password 

251 

252 Examples: 

253 >>> from mcpgateway.db import SessionLocal 

254 >>> import asyncio 

255 >>> with SessionLocal() as db: 

256 ... service = PasswordPolicyService(db) 

257 ... # Note: This is an async method, use in async context 

258 ... # asyncio.run(service.check_password_history("alice@example.com", "NewP@ssw0rd!")) 

259 ... True 

260 True 

261 """ 

262 # First check if new password matches current password 

263 if current_password_hash and await self.password_service.verify_password_async(new_password, current_password_hash): 

264 raise PasswordPolicyError("New password must be different from current password") 

265 

266 # Get user's password history 

267 history_entries = self.db.query(PasswordHistory).filter(PasswordHistory.user_email == email).order_by(PasswordHistory.changed_at.desc()).limit(history_count).all() 

268 

269 # Check if new password matches any in history 

270 for entry in history_entries: 

271 if await self.password_service.verify_password_async(new_password, entry.password_hash): 

272 raise PasswordPolicyError(f"Password was used recently. Please choose a different password " f"(last {history_count} passwords cannot be reused)") 

273 

274 return True

306 def generate_secure_password(length: int = 20, for_service_account: bool = False) -> str: 

307 """Generate cryptographically secure random password. 

308 

309 Args: 

310 length: Password length (default: 20, minimum: 12) 

311 for_service_account: If True, generates UUID-based password (128-bit entropy) 

312 

313 Returns: 

314 str: Randomly generated password 

315 

316 Examples: 

317 >>> from mcpgateway.db import SessionLocal 

318 >>> with SessionLocal() as db: 

319 ... service = PasswordPolicyService(db) 

320 ... password = service.generate_secure_password(20) 

321 ... len(password) >= 20 

322 True 

323 >>> # Service account password (UUID-based, 128-bit entropy) 

324 >>> service_password = service.generate_secure_password(for_service_account=True) 

325 >>> len(service_password) == 36 # UUID format 

326 True 

327 """ 

328 if for_service_account: 

329 # Use UUID for service accounts (128-bit entropy as recommended) 

330 return str(uuid.uuid4()) 

331 

332 # For user accounts, use full printable keyboard characters 

333 # Each character provides ~6.5 bits of entropy 

334 min_length = max(length, 12) 

335 

336 # Use all printable ASCII characters for maximum entropy 

337 charset = string.ascii_letters + string.digits + string.punctuation 

338 

339 # Generate random password 

340 password = "".join(secrets.choice(charset) for _ in range(min_length)) 

341 

342 return password

344 @staticmethod 

345 def _has_sequential_chars(password: str, max_sequential: int = 3) -> bool: 

346 """Check if password contains too many sequential characters. 

347 

348 Args: 

349 password: Password to check 

350 max_sequential: Maximum allowed sequential characters 

351 

352 Returns: 

353 bool: True if password has too many sequential characters 

354 """ 

355 # Check for sequential numbers (e.g., "123", "456") and reverse (e.g., "321", "987") 

356 for i in range(len(password) - max_sequential + 1): 

357 substr = password[i : i + max_sequential] 

358 if substr.isdigit(): 

359 digits = [int(c) for c in substr] 

360 is_ascending = all(digits[j] + 1 == digits[j + 1] for j in range(len(digits) - 1)) 

361 is_descending = all(digits[j] - 1 == digits[j + 1] for j in range(len(digits) - 1)) 

362 if is_ascending or is_descending: 

363 return True 

364 

365 # Check for sequential letters (e.g., "abc", "xyz") and reverse (e.g., "cba", "zyx") 

366 for i in range(len(password) - max_sequential + 1): 

367 substr = password[i : i + max_sequential].lower() 

368 if substr.isalpha(): 

369 is_ascending = all(ord(substr[j]) + 1 == ord(substr[j + 1]) for j in range(len(substr) - 1)) 

370 is_descending = all(ord(substr[j]) - 1 == ord(substr[j + 1]) for j in range(len(substr) - 1)) 

371 if is_ascending or is_descending: 

372 return True 

373 

374 return False

377 def _has_sufficient_entropy(password: str, min_unique_chars: int = 10) -> bool: 

378 """Check if password has sufficient entropy (randomness). 

379 

380 Args: 

381 password: Password to check 

382 min_unique_chars: Minimum number of unique characters required 

383 

384 Returns: 

385 bool: True if password appears to have sufficient entropy 

386 """ 

387 # Check for minimum unique characters 

388 unique_chars = len(set(password)) 

389 if unique_chars < min_unique_chars: 

390 return False 

391 

392 # Check for repeated patterns 

393 if len(password) >= 4: 

394 for i in range(len(password) - 3): 

395 pattern = password[i : i + 2] 

396 if password.count(pattern) > 2: 

397 return False 

398 

399 return True

401 def get_password_strength_score(self, password: str) -> dict: 

402 """Calculate password strength score and provide feedback. 

403 

404 Args: 

405 password: Password to evaluate 

406 

407 Returns: 

408 dict: Score (0-100) and feedback messages 

409 

410 Examples: 

411 >>> from mcpgateway.db import SessionLocal 

412 >>> with SessionLocal() as db: 

413 ... service = PasswordPolicyService(db) 

414 ... result = service.get_password_strength_score("weak") 

415 ... result['score'] < 50 

416 True 

417 """ 

418 score = 0 

419 feedback = [] 

420 

421 # Length scoring 

422 if len(password) >= 12: 

423 score += 25 

424 elif len(password) >= 8: 

425 score += 15 

426 feedback.append("Consider using at least 12 characters") 

427 else: 

428 feedback.append("Password is too short (minimum 12 characters recommended)") 

429 

430 # Complexity scoring 

431 has_lower = bool(re.search(r"[a-z]", password)) 

432 has_upper = bool(re.search(r"[A-Z]", password)) 

433 has_digit = bool(re.search(r"[0-9]", password)) 

434 has_special = bool(re.search(r'[!@#$%^&*(),.?":{}|<>\-_=+\[\]\\;\'`~]', password)) 

435 

436 complexity_count = sum([has_lower, has_upper, has_digit, has_special]) 

437 score += complexity_count * 15 

438 

439 if complexity_count < 3: 

440 feedback.append("Add more character types (uppercase, lowercase, numbers, special)") 

441 

442 # Check for common passwords 

443 if password.lower() not in COMMON_PASSWORDS: 

444 score += 20 

445 else: 

446 feedback.append("Password is too common") 

447 score = min(score, 30) # Cap score for common passwords 

448 

449 # Check for sequential characters 

450 if not self._has_sequential_chars(password): 

451 score += 10 

452 else: 

453 feedback.append("Avoid sequential characters (e.g., '123', 'abc')") 

454 

455 # Entropy check 

456 if self._has_sufficient_entropy(password): 

457 score += 15 

458 else: 

459 feedback.append("Password appears predictable - use more random characters") 

460 

461 return { 

462 "score": min(score, 100), 

463 "feedback": feedback, 

464 "strength": "strong" if score >= 80 else "medium" if score >= 60 else "weak", 

465 }
