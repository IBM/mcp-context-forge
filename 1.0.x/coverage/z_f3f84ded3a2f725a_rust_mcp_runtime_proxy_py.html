39# Hex-only on purpose: server IDs are uuid4().hex (32 hex chars). Non-hex 

40# segments (e.g. "ndh45", "my-server") will never match here and instead fall 

41# through to the _SERVER_SCOPED_PATH_RE defense-in-depth guard, which rejects 

42# them without a database round-trip. 

43_SERVER_ID_RE = re.compile(r"/servers/(?P<server_id>[a-fA-F0-9\-]+)/mcp/?$") 

44# Pattern that detects a server-scoped MCP path even when _SERVER_ID_RE doesn't 

45# match (e.g. empty segment: /servers//mcp). Used as a defense-in-depth guard. 

46_SERVER_SCOPED_PATH_RE = re.compile(r"^/servers/.*/mcp(?:/)?$") 

47_CONTEXTFORGE_SERVER_ID_HEADER = "x-contextforge-server-id" 

48_CONTEXTFORGE_AUTH_CONTEXT_HEADER = "x-contextforge-auth-context" 

49_CONTEXTFORGE_AFFINITY_FORWARDED_HEADER = "x-contextforge-affinity-forwarded" 

50_CLIENT_ERROR_DETAIL = "See server logs" 

51_REQUEST_HOP_BY_HOP_HEADERS = frozenset({"host", "content-length", "connection", "transfer-encoding", "keep-alive"}) 

52_FORWARDED_CHAIN_HEADERS = frozenset({"forwarded", "x-forwarded-for", "x-forwarded-host", "x-forwarded-port", "x-forwarded-proto"}) 

53_INTERNAL_ONLY_REQUEST_HEADERS = frozenset( 

54 { 

55 "x-forwarded-internally", 

56 "x-original-worker", 

57 "x-mcp-session-id", 

58 "x-contextforge-mcp-runtime", 

59 _CONTEXTFORGE_SERVER_ID_HEADER, 

60 _CONTEXTFORGE_AUTH_CONTEXT_HEADER, 

61 _CONTEXTFORGE_AFFINITY_FORWARDED_HEADER, 

62 } 

63) 

64_RESPONSE_HOP_BY_HOP_HEADERS = frozenset({"connection", "transfer-encoding", "keep-alive"})

70async def _validate_server_id(match: re.Match[str] | None, path: str, scope: Scope, receive: Receive, send: Send) -> str | object | None: 

71 """Validate and resolve the server_id from the request path. 

72 

73 Args: 

74 match: Result of ``_SERVER_ID_RE.search(path)``. 

75 path: Original request path. 

76 scope: ASGI scope dict. 

77 receive: ASGI receive callable. 

78 send: ASGI send callable. 

79 

80 Returns: 

81 The validated server_id string, ``None`` when the path is 

82 not server-scoped (legitimate global ``/mcp``), or the 

83 sentinel ``_REJECT`` when an error response has already been 

84 sent and the caller should return immediately. 

85 """ 

86 if match: 

87 server_id = match.group("server_id") 

88 # SECURITY: Validate that the server_id exists in the database 

89 # to prevent unauthorized access via invalid server IDs. 

90 try: 

91 with fresh_db_session() as db: 

92 exists = db.execute(sa_exists().where(DbServer.id == server_id)).scalar() 

93 if not exists: 

94 logger.warning("Invalid server ID in Rust proxy MCP request path: %s", server_id) 

95 response = ORJSONResponse({"detail": "Server not found"}, status_code=404) 

96 await response(scope, receive, send) 

97 return _REJECT 

98 except Exception as e: 

99 logger.error("Failed to validate server ID %s in Rust proxy: %s", server_id, e) 

100 response = ORJSONResponse({"detail": "Service unavailable — unable to verify server"}, status_code=503) 

101 await response(scope, receive, send) 

102 return _REJECT 

103 return server_id 

104 

105 # SECURITY (defense-in-depth): If the path looks server-scoped but 

106 # the primary regex didn't capture a server_id (e.g. empty segment 

107 # /servers//mcp, or an encoding edge case), reject immediately 

108 # rather than falling through to unscoped global behaviour (#3891). 

109 if _SERVER_SCOPED_PATH_RE.search(path): 

110 logger.warning("Server-scoped MCP path with unparseable server ID rejected in Rust proxy: %s", path) 

111 response = ORJSONResponse({"detail": "Invalid server identifier"}, status_code=404) 

112 await response(scope, receive, send) 

113 return _REJECT 

114 

115 return None # Legitimate unscoped /mcp path

118class RustMCPRuntimeProxy: 

119 """Proxy MCP transport traffic to the experimental Rust runtime."""

121 def __init__(self, python_fallback_app) -> None: 

122 """Initialize the proxy with the existing Python MCP transport fallback. 

123 

124 Args: 

125 python_fallback_app: Python MCP transport app used when Rust cannot handle 

126 the request. 

127 """ 

128 self.python_fallback_app = python_fallback_app 

129 self._uds_client: httpx.AsyncClient | None = None 

130 self._uds_client_lock = asyncio.Lock()

132 async def handle_streamable_http(self, scope: Scope, receive: Receive, send: Send) -> None: 

133 """Route MCP transport requests to the Rust runtime and preserve Python fallback for others. 

134 

135 Args: 

136 scope: Incoming ASGI scope. 

137 receive: ASGI receive callable. 

138 send: ASGI send callable. 

139 """ 

140 if scope.get("type") != "http": 

141 logger.debug("Rust MCP runtime deferring to Python fallback: scope type %r is not 'http'", scope.get("type")) 

142 await self.python_fallback_app(scope, receive, send) 

143 return 

144 

145 method = str(scope.get("method", "GET")).upper() 

146 if method not in {"GET", "POST", "DELETE"}: 

147 logger.debug("Rust MCP runtime deferring to Python fallback: HTTP method %r is not supported", method) 

148 await self.python_fallback_app(scope, receive, send) 

149 return 

150 

151 modified_path = str(scope.get("modified_path") or scope.get("path") or "") 

152 match = _SERVER_ID_RE.search(modified_path) 

153 validated = await _validate_server_id(match, modified_path, scope, receive, send) 

154 if validated is _REJECT: 

155 return # Error response already sent 

156 

157 target_url = _build_runtime_mcp_url(scope) 

158 headers = _build_forward_headers(scope) 

159 timeout = httpx.Timeout(settings.experimental_rust_mcp_runtime_timeout_seconds) 

160 

161 try: 

162 client = await self._get_runtime_client() 

163 async with client.stream( 

164 method, 

165 target_url, 

166 content=_stream_request_body(receive) if method == "POST" else b"", 

167 headers=headers, 

168 timeout=timeout, 

169 follow_redirects=False, 

170 ) as response: 

171 await send( 

172 { 

173 "type": "http.response.start", 

174 "status": response.status_code, 

175 "headers": [(name, value) for name, value in response.headers.raw if name.decode("latin-1").lower() not in _RESPONSE_HOP_BY_HOP_HEADERS], 

176 } 

177 ) 

178 async for chunk in response.aiter_bytes(): 

179 if chunk: 

180 await send({"type": "http.response.body", "body": chunk, "more_body": True}) 

181 await send({"type": "http.response.body", "body": b"", "more_body": False}) 

182 except httpx.HTTPError as exc: 

183 logger.error("Experimental Rust MCP runtime request failed: %s", exc) 

184 error_response = ORJSONResponse( 

185 status_code=502, 

186 content={ 

187 "jsonrpc": "2.0", 

188 "id": None, 

189 "error": { 

190 "code": -32000, 

191 "message": "Experimental Rust MCP runtime unavailable", 

192 "data": _CLIENT_ERROR_DETAIL, 

193 }, 

194 }, 

195 ) 

196 await error_response(scope, receive, send) 

197 return

199 async def _get_runtime_client(self) -> httpx.AsyncClient: 

200 """Return the client used for Python -> Rust runtime proxying. 

201 

202 Returns: 

203 An async HTTP client configured for either UDS or loopback HTTP. 

204 """ 

205 uds_path = settings.experimental_rust_mcp_runtime_uds 

206 if not uds_path: 

207 return await get_http_client() 

208 

209 if self._uds_client is not None: 

210 return self._uds_client 

211 

212 async with self._uds_client_lock: 

213 if self._uds_client is None: 

214 self._uds_client = httpx.AsyncClient( 

215 transport=httpx.AsyncHTTPTransport(uds=uds_path), 

216 limits=get_http_limits(), 

217 timeout=httpx.Timeout(settings.experimental_rust_mcp_runtime_timeout_seconds), 

218 follow_redirects=False, 

219 ) 

220 return self._uds_client

223async def _stream_request_body(receive: Receive): 

224 """Yield ASGI request body chunks without buffering the full request. 

225 

226 Args: 

227 receive: ASGI receive callable for the current request. 

228 

229 Yields: 

230 Raw request body chunks as they arrive from the client. 

231 """ 

232 while True: 

233 message = await receive() 

234 if message["type"] == "http.disconnect": 

235 return 

236 if message["type"] != "http.request": 

237 continue 

238 body = message.get("body", b"") 

239 if body: 

240 yield body 

241 if not message.get("more_body", False): 

242 return

245def _extract_server_id_from_scope(scope: Scope) -> str | None: 

246 """Extract server_id when the mounted MCP path came from /servers/<id>/mcp. 

247 

248 This function only extracts the ID from the path. Server existence 

249 validation must be performed separately via _validate_server_id() before 

250 forwarding requests. 

251 

252 Args: 

253 scope: Incoming ASGI scope. 

254 

255 Returns: 

256 The matched server id, or ``None`` when the request is not server-scoped. 

257 """ 

258 modified_path = str(scope.get("modified_path") or scope.get("path") or "") 

259 match = _SERVER_ID_RE.search(modified_path) 

260 return match.group("server_id") if match else None

263def _build_runtime_mcp_url(scope: Scope) -> str: 

264 """Build the target Rust runtime /mcp URL, preserving the query string. 

265 

266 Args: 

267 scope: Incoming ASGI scope. 

268 

269 Returns: 

270 Absolute URL for the Rust sidecar MCP endpoint. 

271 """ 

272 base = urlsplit(settings.experimental_rust_mcp_runtime_url) 

273 query_string = scope.get("query_string", b"") 

274 query = query_string.decode("latin-1") if isinstance(query_string, (bytes, bytearray)) else str(query_string or "") 

275 base_path = base.path.rstrip("/") 

276 if not base_path: 

277 target_path = "/mcp/" 

278 elif base_path.endswith("/mcp"): 

279 target_path = f"{base_path}/" 

280 else: 

281 target_path = f"{base_path}/mcp/" 

282 merged_query = "&".join(part for part in (base.query, query) if part) 

283 return urlunsplit((base.scheme, base.netloc, target_path, merged_query, ""))

286def _build_forward_headers(scope: Scope) -> list[tuple[str, str]]: 

287 """Forward request headers needed by the Rust runtime while stripping internal-only headers. 

288 

289 Args: 

290 scope: Incoming ASGI scope. 

291 

292 Returns: 

293 Header tuples safe to forward to the Rust sidecar. 

294 """ 

295 headers: list[tuple[str, str]] = [] 

296 for item in scope.get("headers") or []: 

297 if not isinstance(item, (tuple, list)) or len(item) != 2: 

298 continue 

299 name, value = item 

300 if not isinstance(name, (bytes, bytearray)) or not isinstance(value, (bytes, bytearray)): 

301 continue 

302 header_name = name.decode("latin-1").lower() 

303 if header_name in _REQUEST_HOP_BY_HOP_HEADERS or header_name in _FORWARDED_CHAIN_HEADERS or header_name in _INTERNAL_ONLY_REQUEST_HEADERS: 

304 continue 

305 headers.append((header_name, value.decode("latin-1"))) 

306 

307 server_id = _extract_server_id_from_scope(scope) 

308 if server_id: 

309 headers.append((_CONTEXTFORGE_SERVER_ID_HEADER, server_id)) 

310 

311 auth_context = _build_forwarded_auth_context_header() 

312 if auth_context is not None: 

313 headers.append((_CONTEXTFORGE_AUTH_CONTEXT_HEADER, auth_context)) 

314 

315 client = scope.get("client") 

316 client_host = client[0] if isinstance(client, (tuple, list)) and client else None 

317 from_loopback = client_host in ("127.0.0.1", "::1") 

318 incoming_headers = { 

319 name.decode("latin-1").lower(): value.decode("latin-1") 

320 for item in scope.get("headers") or [] 

321 if isinstance(item, (tuple, list)) and len(item) == 2 

322 for name, value in [item] 

323 if isinstance(name, (bytes, bytearray)) and isinstance(value, (bytes, bytearray)) 

324 } 

325 if from_loopback and incoming_headers.get("x-forwarded-internally") == "true": 

326 headers.append((_CONTEXTFORGE_AFFINITY_FORWARDED_HEADER, "rust")) 

327 

328 return headers

331def _build_forwarded_auth_context_header() -> str | None: 

332 """Serialize the authenticated MCP context for the trusted internal Python dispatcher. 

333 

334 Returns: 

335 Base64url-encoded auth context for trusted internal forwarding, or ``None`` 

336 when no MCP auth context is available. 

337 """ 

338 auth_context = get_streamable_http_auth_context() 

339 if not auth_context: 

340 return None 

341 encoded = base64.urlsafe_b64encode(orjson.dumps(auth_context)).decode("ascii") 

342 return encoded.rstrip("=")
