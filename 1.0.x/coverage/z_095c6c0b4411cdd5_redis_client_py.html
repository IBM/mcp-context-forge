43def _validate_ssl_settings(settings: Any) -> None: 

44 """Validate Redis SSL file paths and cert content before the client is created. 

45 

46 Raises: 

47 ValueError: On any misconfiguration — missing files, unparseable certs, 

48 or incomplete mTLS pair (certfile without keyfile or vice versa). 

49 """ 

50 errors: list[str] = [] 

51 

52 for attr, label in [ 

53 ("redis_ssl_ca_certs", "CA certificate (REDIS_SSL_CA_CERTS)"), 

54 ("redis_ssl_certfile", "client certificate (REDIS_SSL_CERTFILE)"), 

55 ("redis_ssl_keyfile", "private key (REDIS_SSL_KEYFILE)"), 

56 ]: 

57 path = getattr(settings, attr, None) 

58 if path and not os.path.isfile(path): 

59 errors.append(f"{label} file not found: {path!r}") 

60 

61 if errors: 

62 raise ValueError("Redis SSL misconfiguration:\n" + "\n".join(f" - {e}" for e in errors)) 

63 

64 # Verify CA bundle is parseable 

65 ca_certs = getattr(settings, "redis_ssl_ca_certs", None) 

66 if ca_certs and os.path.isfile(ca_certs): 

67 try: 

68 _ssl.create_default_context(cafile=ca_certs) 

69 except (_ssl.SSLError, OSError) as exc: 

70 raise ValueError(f"Invalid CA certificate {ca_certs!r}: {exc}") from exc 

71 

72 # Verify client cert and key load cleanly when both are provided 

73 certfile = getattr(settings, "redis_ssl_certfile", None) 

74 keyfile = getattr(settings, "redis_ssl_keyfile", None) 

75 if certfile and keyfile and os.path.isfile(certfile) and os.path.isfile(keyfile): 

76 try: 

77 ctx = _ssl.SSLContext(_ssl.PROTOCOL_TLS_CLIENT) 

78 ctx.load_cert_chain(certfile, keyfile) 

79 except (_ssl.SSLError, OSError) as exc: 

80 raise ValueError(f"Invalid client certificate/key ({certfile!r}, {keyfile!r}): {exc}") from exc

83def _build_ssl_kwargs(settings: Any) -> dict[str, Any]: 

84 """Build SSL keyword arguments for Redis TLS connections (redis-py 7.x). 

85 

86 redis-py 7.x SSLConnection accepts individual ssl_* params, not an SSLContext object. 

87 Returns an empty dict when TLS is disabled so callers can unconditionally 

88 do ``connection_kwargs.update(_build_ssl_kwargs(settings))``. 

89 

90 Raises: 

91 ValueError: If REDIS_SSL=true but cert paths are missing, files not found, 

92 or cert content is unparseable. Callers should treat this as a 

93 fatal startup error. 

94 

95 Args: 

96 settings: Application settings instance. 

97 

98 Returns: 

99 Dict of ssl_* kwargs to spread into Redis.from_url() / aioredis.from_url(). 

100 """ 

101 if not settings.redis_ssl: 

102 return {} 

103 

104 _validate_ssl_settings(settings) 

105 

106 kwargs: dict[str, Any] = {} 

107 

108 if settings.redis_ssl_ca_certs: 

109 kwargs["ssl_ca_certs"] = settings.redis_ssl_ca_certs 

110 if settings.redis_ssl_certfile: 

111 kwargs["ssl_certfile"] = settings.redis_ssl_certfile 

112 if settings.redis_ssl_keyfile: 

113 kwargs["ssl_keyfile"] = settings.redis_ssl_keyfile 

114 

115 if not settings.redis_ssl_check_hostname: 

116 # Skip server-cert hostname verification for self-signed certs 

117 kwargs["ssl_cert_reqs"] = "none" 

118 kwargs["ssl_check_hostname"] = False 

119 

120 return kwargs

214def _get_async_parser_class(parser_setting: str) -> tuple[Any, str]: 

215 """Get the appropriate async Redis parser class based on settings. 

216 

217 Args: 

218 parser_setting: One of "auto", "hiredis", or "python" 

219 

220 Returns: 

221 Tuple of (parser_class or None, parser_name) where parser_class is None 

222 for auto-detection (redis-py default behavior) 

223 

224 Raises: 

225 ImportError: If hiredis is required but not available 

226 """ 

227 if parser_setting == "python": 

228 # Force pure-Python async parser 

229 # Third-Party 

230 from redis._parsers import _AsyncRESP2Parser 

231 

232 return _AsyncRESP2Parser, "AsyncRESP2Parser (pure-Python)" 

233 

234 if parser_setting == "hiredis": 

235 # Require hiredis - fail if not available 

236 if not _is_hiredis_available(): 

237 raise ImportError("REDIS_PARSER=hiredis requires hiredis to be installed. " "Install with: pip install 'redis[hiredis]'") 

238 # Don't set parser_class explicitly - let redis-py auto-detect for async 

239 # Setting _AsyncHiredisParser explicitly can cause issues 

240 return None, "AsyncHiredisParser (C extension)" 

241 

242 # "auto" mode - let redis-py auto-detect (prefers hiredis if available) 

243 if _is_hiredis_available(): 

244 return None, "AsyncHiredisParser (C extension, auto-detected)" 

245 return None, "AsyncRESP2Parser (pure-Python, auto-detected)"

248async def get_redis_client() -> Optional[Any]: 

249 """Get or create the shared async Redis client. 

250 

251 Uses hiredis C parser by default for up to 83x faster response parsing. 

252 Parser selection controlled by REDIS_PARSER setting (auto/hiredis/python). 

253 

254 Returns: 

255 Optional[Redis]: Async Redis client, or None if Redis is disabled/unavailable. 

256 

257 Examples: 

258 >>> import asyncio 

259 >>> # When Redis is disabled 

260 >>> async def test_disabled(): 

261 ... from mcpgateway.config import settings 

262 ... original = settings.cache_type 

263 ... settings.cache_type = "memory" 

264 ... from mcpgateway.utils.redis_client import get_redis_client, _reset_client 

265 ... _reset_client() 

266 ... client = await get_redis_client() 

267 ... settings.cache_type = original 

268 ... _reset_client() 

269 ... return client is None 

270 >>> asyncio.run(test_disabled()) 

271 True 

272 """ 

273 global _client, _initialized, _parser_info 

274 

275 if _initialized: 

276 return _client 

277 

278 # First-Party 

279 from mcpgateway.config import settings 

280 

281 if settings.cache_type != "redis" or not settings.redis_url: 

282 logger.info("Redis disabled (cache_type != 'redis' or no redis_url)") 

283 _initialized = True 

284 return None 

285 

286 try: 

287 # Third-Party 

288 import redis.asyncio as aioredis 

289 except ImportError: 

290 logger.warning("redis.asyncio not available, Redis disabled") 

291 _initialized = True 

292 return None 

293 

294 try: 

295 # Get parser configuration (ADR-026) 

296 parser_class, _parser_info = _get_async_parser_class(settings.redis_parser) 

297 

298 # Build connection kwargs 

299 connection_kwargs: dict[str, Any] = { 

300 "decode_responses": settings.redis_decode_responses, 

301 "max_connections": settings.redis_max_connections, 

302 "socket_timeout": settings.redis_socket_timeout, 

303 "socket_connect_timeout": settings.redis_socket_connect_timeout, 

304 "retry_on_timeout": settings.redis_retry_on_timeout, 

305 "health_check_interval": settings.redis_health_check_interval, 

306 "encoding": "utf-8", 

307 "single_connection_client": False, 

308 } 

309 

310 # Only specify parser_class if explicitly set (not auto) 

311 if parser_class is not None: 

312 connection_kwargs["parser_class"] = parser_class 

313 

314 # Warn when URL scheme implies TLS but REDIS_SSL flag is off — our ssl 

315 # settings (CA cert, hostname check) would be silently skipped. 

316 if settings.redis_url and settings.redis_url.startswith("rediss://") and not settings.redis_ssl: 

317 logger.warning("REDIS_URL uses rediss:// scheme but REDIS_SSL=false — " "TLS certificate settings (REDIS_SSL_CA_CERTS, REDIS_SSL_CHECK_HOSTNAME) will not be applied") 

318 

319 # Inject TLS kwargs when REDIS_SSL=true (production). 

320 # Local dev: returns {} → no ssl kwarg → plain TCP connection. 

321 connection_kwargs.update(_build_ssl_kwargs(settings)) 

322 

323 _client = aioredis.from_url(settings.redis_url, **connection_kwargs) 

324 await _client.ping() 

325 logger.info( 

326 f"Redis client initialized: parser={_parser_info}, " 

327 f"pool_size={settings.redis_max_connections}, " 

328 f"timeout={settings.redis_socket_timeout}s, " 

329 f"health_check={settings.redis_health_check_interval}s" 

330 ) 

331 except ImportError as e: 

332 logger.error(f"Redis parser configuration error: {_sanitize(str(e))}") 

333 _client = None 

334 except ValueError as e: 

335 # SSL misconfiguration — bad paths or unparseable certs; treat as fatal 

336 logger.error(f"Redis SSL misconfiguration — client not started: {_sanitize(str(e))}") 

337 _client = None 

338 except Exception as e: 

339 logger.warning(f"Failed to connect to Redis: {_sanitize(str(e))}") 

340 _client = None 

341 

342 _initialized = True 

343 return _client

373async def is_redis_available() -> bool: 

374 """Check if Redis is available and connected. 

375 

376 Returns: 

377 bool: True if Redis is available and responding to ping. 

378 

379 Examples: 

380 >>> import asyncio 

381 >>> async def test_unavailable(): 

382 ... from mcpgateway.config import settings 

383 ... original = settings.cache_type 

384 ... settings.cache_type = "memory"

393 """ 

394 client = await get_redis_client()

399 return True 

400 except Exception: 

401 return False
