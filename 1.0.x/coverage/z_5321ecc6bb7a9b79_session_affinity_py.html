7Multi-worker session affinity for downstream MCP sessions. 

8Keeps each downstream MCP session (identified by its ``Mcp-Session-Id``) 

9pinned to one gateway worker across the horizontal-scale deployment, so 

10the worker-local ``UpstreamSessionRegistry`` can serve subsequent calls 

11without rebuilding upstream state. Per-worker upstream ``ClientSession`` 

12state lives in ``mcpgateway.services.upstream_session_registry``; this 

13module owns the cross-worker affinity layer only. 

14Surface: 

15* Redis-backed ``(downstream_session_id, url, transport, gateway_id)`` → 

16 owning-worker mapping so any worker can look up who owns a session. 

17* Worker heartbeat (``SET EX``) so dead workers can be reclaimed. 

18* Atomic ownership claim via ``SET NX`` and a Lua CAS reclaim script. 

19* Session-owner HTTP/RPC forwarding for cross-worker fanout. 

20* Pub/Sub listener for RPC-style cross-worker requests. 

21* ``is_valid_mcp_session_id`` validation used by the transport layer. 

22"""

71 

72 Subclasses ``RuntimeError`` for backwards compatibility with callers 

73 that still write ``except RuntimeError``. The dedicated type lets the 

74 GET handler distinguish "service genuinely not initialized" from

86class ListenerClaimResult(Enum): 

87 """Outcome of a GET-stream listener claim attempt (ADR-052). 

88 

89 The tri-state lets the GET handler distinguish "another client holds 

90 this session's listener slot" (409 Conflict) from "the storage backing 

91 the claim is unavailable" (503 Service Unavailable). A boolean shape 

92 would force clients to tight-loop on 409 during a Redis outage since 

93 503 is the retryable signal — see ADR-052 § "Single-node vs. 

94 multi-node fallback contract". 

95 """ 

96 

97 WON = "won" 

98 CONFLICT = "conflict" 

99 UNAVAILABLE = "unavailable"

248 async def register_session_mapping( 

249 self, 

250 mcp_session_id: str, 

251 url: str, 

252 gateway_id: str, 

253 transport_type: str, 

254 user_email: Optional[str] = None, 

255 ) -> None: 

256 """Claim ownership of a downstream MCP session in Redis before any request routes to it. 

257 

258 Writes two Redis entries keyed on ``mcp_session_id``: 

259 * ``mcp_session_id → {url, user_hash, identity_hash, transport_type, 

260 gateway_id}`` — used cross-worker to locate the owner of a session. 

261 * ``session_owner:<mcp_session_id> → WORKER_ID`` via ``SET NX`` — 

262 atomically claims this worker as the owner so a second worker 

263 racing the same session doesn't start creating a parallel 

264 upstream connection. 

265 

266 Both entries carry the configured session-affinity TTL and are 

267 refreshed on subsequent calls. Redis failure is non-fatal — same-worker 

268 requests can still route via the owner claim, and a fresh call will 

269 retry. Safe to call repeatedly for an already-owned session. 

270 

271 Args: 

272 mcp_session_id: The downstream MCP session ID from x-mcp-session-id header. 

273 url: The upstream MCP server URL. 

274 gateway_id: The gateway ID. 

275 transport_type: The transport type (sse, streamablehttp). 

276 user_email: The email of the authenticated user (or "system" for unauthenticated). 

277 """ 

278 if not settings.mcpgateway_session_affinity_enabled: 

279 return 

280 

281 # Validate mcp_session_id to prevent Redis key injection 

282 if not self.is_valid_mcp_session_id(mcp_session_id): 

283 logger.warning(f"Invalid mcp_session_id format, skipping session mapping: {mcp_session_id[:20]}...") 

284 return 

285 

286 # Use user email for user_identity, or "anonymous" if not provided 

287 user_identity = user_email or "anonymous" 

288 

289 # Normalize gateway_id to empty string if None for consistent key matching 

290 normalized_gateway_id = gateway_id or "" 

291 

292 # Compute the identity + user hashes used for the Redis mapping value. 

293 identity_hash = hashlib.sha256(mcp_session_id.encode()).hexdigest() 

294 if user_identity == "anonymous": 

295 user_hash = "anonymous" 

296 else: 

297 user_hash = hashlib.sha256(user_identity.encode()).hexdigest() 

298 

299 logger.debug(f"Session affinity pre-registering: {mcp_session_id[:8]}... → {url}, " f"user={SecurityValidator.sanitize_log_message(user_identity)}") 

300 

301 # Store in Redis for multi-worker support AND register ownership atomically 

302 # Registering ownership HERE (during mapping) instead of in acquire() prevents 

303 # a race condition where two workers could both start creating sessions before 

304 # either registers ownership 

305 try: 

306 # First-Party 

307 from mcpgateway.utils.redis_client import ( # pylint: disable=import-outside-toplevel 

308 get_redis_client, 

309 ) 

310 

311 redis = await get_redis_client() 

312 if redis: 

313 redis_key = self._session_mapping_redis_key(mcp_session_id, url, transport_type, normalized_gateway_id) 

314 

315 # Store pool_key as JSON for easy deserialization 

316 pool_key_data = { 

317 "user_hash": user_hash, 

318 "url": url, 

319 "identity_hash": identity_hash, 

320 "transport_type": transport_type, 

321 "gateway_id": normalized_gateway_id, 

322 } 

323 await redis.setex( 

324 redis_key, 

325 settings.mcpgateway_session_affinity_ttl, 

326 orjson.dumps(pool_key_data), 

327 ) # TTL from config 

328 

329 # CRITICAL: Register ownership atomically with mapping. 

330 # This claims ownership BEFORE any session creation attempt, preventing 

331 # the race condition where two workers both start creating sessions 

332 owner_key = self._session_owner_key(mcp_session_id) 

333 # Atomic claim with TTL (avoids the SETNX/EXPIRE crash window). 

334 was_set = await redis.set( 

335 owner_key, 

336 WORKER_ID, 

337 nx=True, 

338 ex=settings.mcpgateway_session_affinity_ttl, 

339 ) 

340 if was_set: 

341 logger.debug(f"Session ownership claimed (SET NX): {mcp_session_id[:8]}... → worker {WORKER_ID}") 

342 else: 

343 # Another worker already claimed ownership 

344 existing_owner = await redis.get(owner_key) 

345 owner_id = existing_owner.decode() if isinstance(existing_owner, bytes) else existing_owner 

346 logger.debug(f"Session ownership already claimed by {owner_id}: {mcp_session_id[:8]}...") 

347 

348 logger.debug(f"Session affinity pre-registered (Redis): {mcp_session_id[:8]}... TTL={settings.mcpgateway_session_affinity_ttl}s") 

349 except Exception as e: 

350 # Redis failure is non-fatal - local mapping still works for same-worker requests 

351 logger.debug(f"Failed to store session mapping in Redis: {e}")

392 # ------------------------------------------------------------------ 

393 # GET-stream listener claims (ADR-052) 

394 # 

395 # Public surface: 

396 # - claim_listener → ListenerClaimResult (Won / Conflict / Unavailable) 

397 # - heartbeat_listener → bool (True = still owner, False = lost) 

398 # - release_listener → bool (True = released, False = not owner / expired) 

399 # 

400 # The tri-state result on claim is the key: Redis-down (Unavailable) and 

401 # someone-else-has-it (Conflict) are different conditions and the GET 

402 # handler responds 503 vs 409 accordingly. ADR-052 calls this out. 

403 # 

404 # The MCP spec mandates one server→client SSE stream per session. The 

405 # claim is held by whichever node accepts the GET; cross-node uniqueness 

406 # uses a Redis SET NX, single-node uses an in-process dict + lock. Same 

407 # API both ways. 

408 # ------------------------------------------------------------------ 

409 

410 # Lua: heartbeat (refresh TTL) only if the caller still owns the claim. 

411 _HEARTBEAT_LISTENER_LUA = "if redis.call('GET', KEYS[1]) == ARGV[1] then redis.call('EXPIRE', KEYS[1], ARGV[2]) return 1 else return 0 end" 

412 # Lua: release only if the caller still owns the claim. 

413 _RELEASE_LISTENER_LUA = "if redis.call('GET', KEYS[1]) == ARGV[1] then redis.call('DEL', KEYS[1]) return 1 else return 0 end"

478 async def claim_listener(self, mcp_session_id: str, connection_id: str) -> "ListenerClaimResult": 

479 """Claim the single GET /mcp listener slot for a session. 

480 

481 Args: 

482 mcp_session_id: Downstream MCP session id. 

483 connection_id: Stable identifier for the GET connection trying 

484 to claim — also the value to present to ``heartbeat_listener`` 

485 and ``release_listener``. 

486 

487 Returns: 

488 ``ListenerClaimResult.WON`` — the caller now holds the claim. 

489 ``ListenerClaimResult.CONFLICT`` — another listener already 

490 holds it (caller should respond 409). 

491 ``ListenerClaimResult.UNAVAILABLE`` — Redis-mode storage failed 

492 (Redis configured but unreachable, eval errored, etc.) — the 

493 caller should respond 503 because the single-listener invariant 

494 cannot be enforced. Also returned for invalid session ids, 

495 since callers shouldn't 409 a malformed input. 

496 """ 

497 if not self.is_valid_mcp_session_id(mcp_session_id): 

498 return ListenerClaimResult.UNAVAILABLE 

499 ttl = settings.mcp_get_stream_listener_ttl_seconds 

500 # Multi-node: Redis SET NX EX is the authority. 

501 if settings.cache_type == "redis": 

502 try: 

503 # First-Party 

504 from mcpgateway.utils.redis_client import ( # pylint: disable=import-outside-toplevel 

505 get_redis_client, 

506 ) 

507 

508 redis = await get_redis_client() 

509 if redis is None: 

510 # cache_type=redis but no client — configuration drift or 

511 # startup race. Treat as unavailable, not conflict. 

512 return ListenerClaimResult.UNAVAILABLE 

513 key = self._listener_claim_key(mcp_session_id) 

514 won = bool(await redis.set(key, connection_id, nx=True, ex=ttl)) 

515 return ListenerClaimResult.WON if won else ListenerClaimResult.CONFLICT 

516 except Exception as exc: 

517 # Redis configured but unreachable / eval failed. Surface 

518 # explicitly so the GET handler returns 503 (per ADR-052), 

519 # not 409 — the previous boolean return collapsed both. 

520 self._log_redis_listener_error("claim", mcp_session_id, exc) 

521 return ListenerClaimResult.UNAVAILABLE 

522 # Single-node: in-process dict guarded by lock gives the same atomicity. 

523 async with self._listener_lock: 

524 self._purge_expired_listener_claims_locked() 

525 existing = self._listener_claims.get(mcp_session_id) 

526 if existing is not None: 

527 return ListenerClaimResult.CONFLICT 

528 self._listener_claims[mcp_session_id] = (connection_id, time.time() + ttl) 

529 return ListenerClaimResult.WON

531 async def heartbeat_listener(self, mcp_session_id: str, connection_id: str) -> bool: 

532 """Refresh a held listener claim. 

533 

534 Args: 

535 mcp_session_id: Downstream MCP session id. 

536 connection_id: The connection id presented at ``claim_listener``. 

537 

538 Returns: 

539 True if the heartbeat refreshed the TTL, False if the claim no 

540 longer belongs to ``connection_id`` (caller should close the 

541 stream). 

542 """ 

543 if not self.is_valid_mcp_session_id(mcp_session_id): 

544 return False 

545 ttl = settings.mcp_get_stream_listener_ttl_seconds 

546 if settings.cache_type == "redis": 

547 try: 

548 # First-Party 

549 from mcpgateway.utils.redis_client import ( # pylint: disable=import-outside-toplevel 

550 get_redis_client, 

551 ) 

552 

553 redis = await get_redis_client() 

554 if redis is None: 

555 return False 

556 key = self._listener_claim_key(mcp_session_id) 

557 result = await redis.eval(self._HEARTBEAT_LISTENER_LUA, 1, key, connection_id, ttl) 

558 return bool(result) 

559 except Exception as exc: 

560 self._log_redis_listener_error("heartbeat", mcp_session_id, exc) 

561 return False 

562 async with self._listener_lock: 

563 # Purge expired claims so a stale entry whose TTL elapsed 

564 # without a release call (worker crashed mid-stream, etc.) 

565 # doesn't read as "still owned by us" — without this purge 

566 # the .get() below would return a tuple whose connection_id 

567 # match still hits even though the slot is logically free. 

568 self._purge_expired_listener_claims_locked() 

569 existing = self._listener_claims.get(mcp_session_id) 

570 if existing is None or existing[0] != connection_id: 

571 return False 

572 self._listener_claims[mcp_session_id] = (connection_id, time.time() + ttl) 

573 return True

586 if not self.is_valid_mcp_session_id(mcp_session_id):

588 if settings.cache_type == "redis":

590 # First-Party 

591 from mcpgateway.utils.redis_client import ( # pylint: disable=import-outside-toplevel 

592 get_redis_client, 

593 ) 

594 

595 redis = await get_redis_client() 

596 if redis is None: 

597 return False 

598 key = self._listener_claim_key(mcp_session_id) 

599 result = await redis.eval(self._RELEASE_LISTENER_LUA, 1, key, connection_id) 

600 return bool(result) 

601 except Exception as exc: 

602 self._log_redis_listener_error("release", mcp_session_id, exc) 

603 return False 

604 async with self._listener_lock: 

605 # Same purge rationale as ``heartbeat_listener``: an 

606 # expired-but-undeleted entry whose connection_id matches 

607 # ours would otherwise return ``True`` from this release 

608 # call as if a real claim was dropped.

611 if existing is None or existing[0] != connection_id: 

612 return False 

613 self._listener_claims.pop(mcp_session_id, None) 

614 return True

660 async def register_session_owner(self, mcp_session_id: str) -> None: 

661 """Claim this worker as the owner of a downstream MCP session, or refresh the existing lease. 

662 

663 Runs a single Lua CAS: ``SET EX`` on miss (first-time claim), ``EXPIRE`` 

664 on hit where the cached owner matches this worker (TTL refresh), no-op 

665 otherwise (another worker already owns it). Callers don't need to 

666 distinguish claim from refresh — the semantics are identical from 

667 their side. Redis failure is non-fatal and logged at debug. 

668 

669 Args: 

670 mcp_session_id: The downstream ``Mcp-Session-Id`` header value. 

671 """ 

672 if not settings.mcpgateway_session_affinity_enabled: 

673 return 

674 

675 if not self.is_valid_mcp_session_id(mcp_session_id): 

676 logger.debug("Invalid mcp_session_id for owner registration, skipping") 

677 return 

678 

679 try: 

680 # First-Party 

681 from mcpgateway.utils.redis_client import ( # pylint: disable=import-outside-toplevel 

682 get_redis_client, 

683 ) 

684 

685 redis = await get_redis_client() 

686 if redis: 

687 key = self._session_owner_key(mcp_session_id) 

688 

689 # Do not steal ownership: only claim if missing, or refresh TTL if we already own. 

690 # Lua keeps this atomic. 

691 script = """ 

692 local cur = redis.call('GET', KEYS[1]) 

693 if not cur then 

694 redis.call('SET', KEYS[1], ARGV[1], 'EX', ARGV[2]) 

695 return 1 

696 end 

697 if cur == ARGV[1] then 

698 redis.call('EXPIRE', KEYS[1], ARGV[2]) 

699 return 2 

700 end 

701 return 0 

702 """ 

703 ttl = int(settings.mcpgateway_session_affinity_ttl) 

704 outcome = await redis.eval(script, 1, key, WORKER_ID, ttl) 

705 logger.debug(f"Owner registration outcome={outcome} for session {mcp_session_id[:8]}...") 

706 except Exception as e: 

707 # Redis failure is non-fatal - single worker mode still works 

708 logger.debug(f"Failed to register session owner in Redis: {e}")

743 async def forward_request_to_owner( 

744 self, 

745 mcp_session_id: str, 

746 request_data: Dict[str, Any], 

747 timeout: Optional[float] = None, 

748 ) -> Optional[Dict[str, Any]]: 

749 """Forward RPC request to the worker that owns the session owner. 

750 

751 This method checks Redis to find which worker owns the session owner for 

752 the given mcp_session_id. If owned by another worker, it forwards the 

753 request via Redis pub/sub and waits for the response. 

754 

755 Args: 

756 mcp_session_id: The MCP session ID from x-mcp-session-id header. 

757 request_data: The RPC request data to forward. 

758 timeout: Optional timeout in seconds (default from config). 

759 

760 Returns: 

761 The response from the owner worker, or None if we own the session 

762 (caller should execute locally) or if Redis is unavailable. 

763 

764 Raises: 

765 asyncio.TimeoutError: If the forwarded request times out. 

766 """ 

767 if not settings.mcpgateway_session_affinity_enabled: 

768 return None 

769 

770 if not self.is_valid_mcp_session_id(mcp_session_id): 

771 return None 

772 

773 effective_timeout = timeout if timeout is not None else settings.mcpgateway_pool_rpc_forward_timeout 

774 

775 try: 

776 # First-Party 

777 from mcpgateway.utils.redis_client import ( # pylint: disable=import-outside-toplevel 

778 get_redis_client, 

779 ) 

780 

781 redis = await get_redis_client() 

782 if not redis: 

783 return None # Execute locally - no Redis 

784 

785 # Check who owns this session 

786 owner = await redis.get(self._session_owner_key(mcp_session_id)) 

787 method = request_data.get("method", "unknown") 

788 if not owner: 

789 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {mcp_session_id[:8]}... | Method: {method} | No owner → execute locally (new session)") 

790 return None # No owner registered - execute locally (new session) 

791 

792 owner_id = owner.decode() if isinstance(owner, bytes) else owner 

793 if owner_id == WORKER_ID: 

794 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {mcp_session_id[:8]}... | Method: {method} | We own it → execute locally") 

795 return None # We own it - execute locally 

796 

797 if not await self._is_worker_alive(owner_id): 

798 logger.warning(f"[AFFINITY] Owner {owner_id} is dead for session {mcp_session_id[:8]}...") 

799 # CAS: reclaim only if still owned by the dead worker 

800 cas_script = """ 

801 local cur = redis.call('GET', KEYS[1]) 

802 if cur == ARGV[1] then 

803 redis.call('SET', KEYS[1], ARGV[2], 'EX', ARGV[3]) 

804 return 1 

805 end 

806 return 0 

807 """ 

808 ttl = int(settings.mcpgateway_session_affinity_ttl) 

809 reclaimed = await redis.eval( 

810 cas_script, 

811 1, 

812 self._session_owner_key(mcp_session_id), 

813 owner_id, 

814 WORKER_ID, 

815 ttl, 

816 ) 

817 if reclaimed == 1: 

818 logger.info(f"[AFFINITY] Reclaimed session {mcp_session_id[:8]}... from dead worker {owner_id} → execute locally") 

819 return None # We won the reclaim - execute locally 

820 # Another worker already reclaimed; re-read the new owner and forward 

821 new_owner = await redis.get(self._session_owner_key(mcp_session_id)) 

822 if not new_owner: 

823 return None # Key vanished - execute locally 

824 owner_id = new_owner.decode() if isinstance(new_owner, bytes) else new_owner 

825 if owner_id == WORKER_ID: 

826 return None # We ended up as owner 

827 logger.info(f"[AFFINITY] Session {mcp_session_id[:8]}... reclaimed by {owner_id} → forwarding to new owner") 

828 

829 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {mcp_session_id[:8]}... | Method: {method} | Owner: {owner_id} → forwarding") 

830 

831 # Forward to owner worker via pub/sub 

832 response_id = str(uuid.uuid4()) 

833 response_channel = f"mcpgw:pool_rpc_response:{response_id}" 

834 

835 # Subscribe to response channel 

836 async with redis.pubsub() as pubsub: 

837 await pubsub.subscribe(response_channel) 

838 

839 try: 

840 # Prepare request with response channel 

841 forward_data = { 

842 "type": "rpc_forward", 

843 **request_data, 

844 "response_channel": response_channel, 

845 "mcp_session_id": mcp_session_id, 

846 } 

847 

848 # Publish request to owner's channel 

849 await redis.publish(f"mcpgw:pool_rpc:{owner_id}", orjson.dumps(forward_data)) 

850 self._forwarded_requests += 1 

851 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {mcp_session_id[:8]}... | Method: {method} | Published to worker {owner_id}") 

852 

853 # Wait for response 

854 async with asyncio.timeout(effective_timeout): 

855 async for msg in pubsub.listen(): 

856 if msg["type"] == "message": 

857 return orjson.loads(msg["data"]) 

858 finally: 

859 await pubsub.unsubscribe(response_channel) 

860 

861 except asyncio.TimeoutError: 

862 self._forwarded_request_timeouts += 1 

863 logger.warning(f"Timeout forwarding request to owner for session {mcp_session_id[:8]}...") 

864 raise 

865 except Exception as e: 

866 self._forwarded_request_failures += 1 

867 logger.debug(f"Error forwarding request to owner: {e}") 

868 return None # Execute locally on error

870 async def start_rpc_listener(self) -> None: 

871 """Start listening for forwarded RPC and HTTP requests on this worker's channels. 

872 

873 This method subscribes to Redis pub/sub channels specific to this worker 

874 and processes incoming forwarded requests from other workers: 

875 - mcpgw:pool_rpc:{WORKER_ID} - for SSE transport JSON-RPC forwards 

876 - mcpgw:pool_http:{WORKER_ID} - for Streamable HTTP request forwards 

877 """ 

878 if not settings.mcpgateway_session_affinity_enabled: 

879 return 

880 

881 try: 

882 # First-Party 

883 from mcpgateway.utils.redis_client import ( # pylint: disable=import-outside-toplevel 

884 get_redis_client, 

885 ) 

886 

887 redis = await get_redis_client() 

888 if not redis: 

889 logger.debug("Redis not available, RPC listener not started") 

890 return 

891 

892 rpc_channel = f"mcpgw:pool_rpc:{WORKER_ID}" 

893 http_channel = f"mcpgw:pool_http:{WORKER_ID}" 

894 async with redis.pubsub() as pubsub: 

895 await pubsub.subscribe(rpc_channel, http_channel) 

896 logger.info(f"RPC/HTTP listener started for worker {WORKER_ID} on channels: {rpc_channel}, {http_channel}") 

897 

898 try: 

899 while not self._closed: 

900 try: 

901 msg = await pubsub.get_message(ignore_subscribe_messages=True, timeout=1.0) 

902 if msg and msg["type"] == "message": 

903 request = orjson.loads(msg["data"]) 

904 forward_type = request.get("type") 

905 response_channel = request.get("response_channel") 

906 

907 if response_channel: 

908 if forward_type == "rpc_forward": 

909 # Execute forwarded RPC request for SSE transport 

910 response = await self._execute_forwarded_request(request) 

911 await redis.publish(response_channel, orjson.dumps(response)) 

912 logger.debug(f"Processed forwarded RPC request, response sent to {response_channel}") 

913 elif forward_type == "http_forward": 

914 # Execute forwarded HTTP request for Streamable HTTP transport 

915 await self._execute_forwarded_http_request(request, redis) 

916 else: 

917 logger.warning(f"Unknown forward type: {forward_type}") 

918 except Exception as e: 

919 logger.warning(f"Error processing forwarded request: {e}") 

920 finally: 

921 await pubsub.unsubscribe(rpc_channel, http_channel) 

922 logger.info(f"RPC/HTTP listener stopped for worker {WORKER_ID}") 

923 

924 except Exception as e: 

925 logger.warning(f"RPC/HTTP listener failed: {e}")

927 async def _execute_forwarded_request(self, request: Dict[str, Any]) -> Dict[str, Any]: 

928 """Execute a forwarded RPC request locally via internal HTTP call. 

929 

930 This method handles RPC requests that were forwarded from another worker. 

931 Instead of handling specific methods here, we make an internal HTTP call 

932 to the local /rpc endpoint which reuses ALL existing method handling logic. 

933 

934 The x-forwarded-internally header prevents infinite forwarding loops. 

935 

936 Args: 

937 request: The forwarded RPC request containing method, params, headers, req_id, etc. 

938 

939 Returns: 

940 The JSON-RPC response from the local endpoint. 

941 """ 

942 try: 

943 method = request.get("method") 

944 params = request.get("params", {}) 

945 headers = request.get("headers", {}) 

946 req_id = request.get("req_id", 1) 

947 mcp_session_id = request.get("mcp_session_id", "unknown") 

948 session_short = mcp_session_id[:8] if len(mcp_session_id) >= 8 else mcp_session_id 

949 

950 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {session_short}... | Method: {method} | Received forwarded request, executing locally") 

951 

952 # Make internal HTTP/HTTPS call to local /rpc endpoint. 

953 # This reuses ALL existing method handling logic without duplication. 

954 internal_base_url = internal_loopback_base_url() 

955 async with httpx.AsyncClient(verify=internal_loopback_verify()) as client: 

956 # Build headers for internal request - forward original headers 

957 # but add x-forwarded-internally to prevent infinite loops. 

958 # Relies on the originating transport having already filtered 

959 # passthrough headers via extract_headers_for_loopback (#3640). 

960 internal_headers = dict(headers) 

961 internal_headers["x-forwarded-internally"] = "true" 

962 # Ensure content-type is set 

963 internal_headers["content-type"] = "application/json" 

964 

965 response = await client.post( 

966 f"{internal_base_url}/rpc", 

967 json={ 

968 "jsonrpc": "2.0", 

969 "method": method, 

970 "params": params, 

971 "id": req_id, 

972 }, 

973 headers=internal_headers, 

974 timeout=settings.mcpgateway_pool_rpc_forward_timeout, 

975 ) 

976 

977 # Gate on HTTP status first: non-2xx responses are errors 

978 # even if the body parses as JSON. 

979 if not response.is_success: 

980 try: 

981 response_data = response.json() 

982 except ValueError: 

983 response_data = {} 

984 if not isinstance(response_data, dict): 

985 response_data = {} 

986 

987 # If body is a JSON-RPC error ({"error": {...}}), propagate it 

988 if "error" in response_data and isinstance(response_data["error"], dict): 

989 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {session_short}... | Method: {method} | Forwarded execution completed with error (HTTP {response.status_code})") 

990 return {"error": response_data["error"]} 

991 

992 # Non-JSON-RPC error body (e.g. {"detail": "..."}): map to JSON-RPC error 

993 detail = response_data.get("detail", response.text[:200] or "Unknown error") 

994 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {session_short}... | Method: {method} | Forwarded execution failed with HTTP {response.status_code}") 

995 return { 

996 "error": { 

997 "code": -32603, 

998 "message": f"Forwarded request failed (HTTP {response.status_code}): {detail}", 

999 } 

1000 } 

1001 

1002 # Parse successful response 

1003 response_data = response.json() 

1004 

1005 # Extract result or error from JSON-RPC response 

1006 if "error" in response_data: 

1007 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {session_short}... | Method: {method} | Forwarded execution completed with error") 

1008 return {"error": response_data["error"]} 

1009 logger.info(f"[AFFINITY] Worker {WORKER_ID} | Session {session_short}... | Method: {method} | Forwarded execution completed successfully") 

1010 return {"result": response_data.get("result", {})} 

1011 

1012 except httpx.TimeoutException: 

1013 logger.warning(f"Timeout executing forwarded request: {request.get('method')}") 

1014 return {"error": {"code": -32603, "message": "Internal request timeout"}} 

1015 except Exception as e: 

1016 logger.warning(f"Error executing forwarded request: {e}") 

1017 return {"error": {"code": -32603, "message": str(e)}}

1019 async def _execute_forwarded_http_request(self, request: Dict[str, Any], redis: Any) -> None: 

1020 """Execute a forwarded HTTP request locally and return response via Redis. 

1021 

1022 This method handles full HTTP requests forwarded from other workers for 

1023 Streamable HTTP transport session affinity. It reconstructs the HTTP request, 

1024 makes an internal call to the appropriate endpoint, and publishes the response 

1025 back through Redis. 

1026 

1027 Args: 

1028 request: Serialized HTTP request data from Redis Pub/Sub containing: 

1029 - type: "http_forward" 

1030 - response_channel: Redis channel to publish response to 

1031 - mcp_session_id: Session identifier 

1032 - method: HTTP method (GET, POST, DELETE) 

1033 - path: Request path (e.g., /mcp) 

1034 - query_string: Query parameters 

1035 - headers: Request headers dict 

1036 - body: Hex-encoded request body 

1037 redis: Redis client for publishing response 

1038 """ 

1039 response_channel = None 

1040 try: 

1041 response_channel = request.get("response_channel") 

1042 method = request.get("method") 

1043 path = request.get("path") 

1044 query_string = request.get("query_string", "") 

1045 headers = request.get("headers", {}) 

1046 body_hex = request.get("body", "") 

1047 mcp_session_id = request.get("mcp_session_id") 

1048 

1049 # Decode hex body back to bytes 

1050 body = bytes.fromhex(body_hex) if body_hex else b"" 

1051 

1052 session_short = mcp_session_id[:8] if mcp_session_id and len(mcp_session_id) >= 8 else "unknown" 

1053 logger.debug(f"[HTTP_AFFINITY] Worker {WORKER_ID} | Session {session_short}... | Received forwarded HTTP request: {method} {path}") 

1054 

1055 # Add internal forwarding headers to prevent loops. 

1056 # Relies on the originating transport having already filtered 

1057 # passthrough headers via extract_headers_for_loopback (#3640). 

1058 internal_headers = dict(headers) 

1059 internal_headers["x-forwarded-internally"] = "true" 

1060 internal_headers["x-original-worker"] = request.get("original_worker", "unknown") 

1061 

1062 # Make internal HTTP/HTTPS request to local endpoint 

1063 url = f"{internal_loopback_base_url()}{path}" 

1064 if query_string: 

1065 url = f"{url}?{query_string}" 

1066 

1067 async with httpx.AsyncClient(verify=internal_loopback_verify()) as client: 

1068 response = await client.request( 

1069 method=method, 

1070 url=url, 

1071 headers=internal_headers, 

1072 content=body, 

1073 timeout=settings.mcpgateway_pool_rpc_forward_timeout, 

1074 ) 

1075 

1076 logger.debug(f"[HTTP_AFFINITY] Worker {WORKER_ID} | Session {session_short}... | Executed locally: {response.status_code}") 

1077 

1078 # Serialize response for Redis transport 

1079 response_data = { 

1080 "status": response.status_code, 

1081 "headers": dict(response.headers), 

1082 "body": response.content.hex(), # Hex encode binary response 

1083 } 

1084 

1085 # Publish response back to requesting worker 

1086 if redis and response_channel: 

1087 await redis.publish(response_channel, orjson.dumps(response_data)) 

1088 logger.debug(f"[HTTP_AFFINITY] Published HTTP response to Redis channel: {response_channel}") 

1089 

1090 except Exception as e: 

1091 logger.error(f"Error executing forwarded HTTP request: {e}") 

1092 # Try to send error response if possible 

1093 if redis and response_channel: 

1094 error_response = { 

1095 "status": 500, 

1096 "headers": {"content-type": "application/json"}, 

1097 "body": orjson.dumps({"error": "Internal forwarding error"}).hex(), 

1098 } 

1099 try: 

1100 await redis.publish(response_channel, orjson.dumps(error_response)) 

1101 except Exception as publish_error: 

1102 logger.debug(f"Failed to publish error response via Redis: {publish_error}")

1153 

1154 session_short = mcp_session_id[:8] if len(mcp_session_id) >= 8 else mcp_session_id 

1155 logger.debug(f"[HTTP_AFFINITY] Worker {WORKER_ID} | Session {session_short}... | {method} {path} | Forwarding to worker {owner_worker_id}") 

1156 

1157 try: 

1158 # First-Party 

1159 from mcpgateway.utils.redis_client import ( # pylint: disable=import-outside-toplevel 

1160 get_redis_client, 

1161 ) 

1162 

1163 redis = await get_redis_client() 

1164 if not redis: 

1165 logger.warning("Redis unavailable for HTTP forwarding, executing locally") 

1166 return None # Fall back to local execution 

1167 

1168 # Generate unique response channel for this request 

1169 response_uuid = uuid.uuid4().hex 

1170 response_channel = f"mcpgw:pool_http_response:{response_uuid}" 

1171 

1172 # Serialize HTTP request for Redis transport 

1173 forward_data = { 

1174 "type": "http_forward", 

1175 "response_channel": response_channel, 

1176 "mcp_session_id": mcp_session_id, 

1177 "method": method, 

1178 "path": path, 

1179 "query_string": query_string, 

1180 "headers": headers, 

1181 "body": body.hex() if body else "", # Hex encode binary body 

1182 "original_worker": WORKER_ID, 

1183 "timestamp": time.time(), 

1184 } 

1185 

1186 # Subscribe to response channel BEFORE publishing request (prevent race) 

1187 async with redis.pubsub() as pubsub: 

1188 await pubsub.subscribe(response_channel) 

1189 

1190 try: 

1191 # Publish forwarded request to owner worker's HTTP channel 

1192 owner_channel = f"mcpgw:pool_http:{owner_worker_id}" 

1193 await redis.publish(owner_channel, orjson.dumps(forward_data)) 

1194 logger.debug(f"[HTTP_AFFINITY] Published HTTP request to Redis channel: {owner_channel}") 

1195 

1196 # Wait for response with timeout 

1197 timeout = settings.mcpgateway_pool_rpc_forward_timeout 

1198 async with asyncio.timeout(timeout): 

1199 while True: 

1200 msg = await pubsub.get_message(ignore_subscribe_messages=True, timeout=0.1) 

1201 if msg and msg["type"] == "message": 

1202 response_data = orjson.loads(msg["data"]) 

1203 logger.debug(f"[HTTP_AFFINITY] Received HTTP response via Redis: status={response_data.get('status')}") 

1204 

1205 # Decode hex body back to bytes 

1206 body_hex = response_data.get("body", "") 

1207 response_data["body"] = bytes.fromhex(body_hex) if body_hex else b"" 

1208 

1209 self._forwarded_requests += 1 

1210 return response_data 

1211 

1212 finally: 

1213 await pubsub.unsubscribe(response_channel) 

1214 

1215 except asyncio.TimeoutError: 

1216 self._forwarded_request_timeouts += 1 

1217 logger.warning(f"Timeout forwarding HTTP request to owner {owner_worker_id}") 

1218 return None 

1219 except Exception as e: 

1220 self._forwarded_request_failures += 1 

1221 logger.warning(f"Error forwarding HTTP request via Redis: {e}") 

1222 return None
